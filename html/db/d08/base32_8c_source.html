93 /* Base32 encode IN array of size INLEN into OUT array of size OUTLEN.
94  If OUTLEN is less than BASE32_LENGTH(INLEN), write as many bytes as
95  possible. If OUTLEN is larger than BASE32_LENGTH(INLEN), also zero
96  terminate the output buffer. */
97 void base32_encode(const char *restrict in, size_t inlen, char *restrict out,
98  size_t outlen) {
99  static const char b32str[32] = "ABCDEFGHIJKLMNOPQRSTUVWXYZ234567";
100 
101  while (inlen && outlen) {
102  *out++ = b32str[(to_uchar(in[0]) >> 3) & 0x1f];
103  if (!--outlen)
104  break;
105  *out++ =
106  b32str[((to_uchar(in[0]) << 2) + (--inlen ? to_uchar(in[1]) >> 6 : 0)) &
107  0x1f];
108  if (!--outlen)
109  break;
110  *out++ = (inlen ? b32str[(to_uchar(in[1]) >> 1) & 0x1f] : '=');
111  if (!--outlen)
112  break;
113  *out++ = (inlen ? b32str[((to_uchar(in[1]) << 4) +
114  (--inlen ? to_uchar(in[2]) >> 4 : 0)) &
115  0x1f]
116  : '=');
117  if (!--outlen)
118  break;
119  *out++ = (inlen ? b32str[((to_uchar(in[2]) << 1) +
120  (--inlen ? to_uchar(in[3]) >> 7 : 0)) &
121  0x1f]
122  : '=');
123  if (!--outlen)
124  break;
125  *out++ = (inlen ? b32str[(to_uchar(in[3]) >> 2) & 0x1f] : '=');
126  if (!--outlen)
127  break;
128  *out++ = (inlen ? b32str[((to_uchar(in[3]) << 3) +
129  (--inlen ? to_uchar(in[4]) >> 5 : 0)) &
130  0x1f]
131  : '=');
132  if (!--outlen)
133  break;
134  *out++ = inlen ? b32str[to_uchar(in[4]) & 0x1f] : '=';
135  if (!--outlen)
136  break;
137  if (inlen)
138  inlen--;
139  if (inlen)
140  in += 5;
141  }
142 
143  if (outlen)
144  *out = '\0';
145 }

147 /* Allocate a buffer and store zero terminated base32 encoded data
148  from array IN of size INLEN, returning BASE32_LENGTH(INLEN), i.e.,
149  the length of the encoded data, excluding the terminating zero. On
150  return, the OUT variable will hold a pointer to newly allocated
151  memory that must be deallocated by the caller. If output string
152  length would overflow, 0 is returned and OUT is set to NULL. If
153  memory allocation failed, OUT is set to NULL, and the return value
154  indicates length of the requested memory block, i.e.,
155  BASE32_LENGTH(inlen) + 1. */
156 size_t base32_encode_alloc(const char *in, size_t inlen, char **out) {
157  size_t outlen = 1 + BASE32_LENGTH(inlen);
158 
159  /* Check for overflow in outlen computation.
160  *
161  * If there is no overflow, outlen >= inlen.
162  *
163  * TODO Is this a sufficient check? (See the notes in base64.c.)
164  */
165  if (inlen > outlen) {
166  *out = NULL;
167  return 0;
168  }
169 
170  *out = malloc(outlen);
171  if (!*out)
172  return outlen;
173 
174  base32_encode(in, inlen, *out, outlen);
175 
176  return outlen - 1;
177 }

294 /* If CTX->i is 0 or 8, there are eight or more bytes in [*IN..IN_END), and
295  none of those eight is a newline, then return *IN. Otherwise, copy up to
296  4 - CTX->i non-newline bytes from that range into CTX->buf, starting at
297  index CTX->i and setting CTX->i to reflect the number of bytes copied,
298  and return CTX->buf. In either case, advance *IN to point to the byte
299  after the last one processed, and set *N_NON_NEWLINE to the number of
300  verified non-newline bytes accessible through the returned pointer. */
301 static char *get_8(struct base32_decode_context *ctx, char const *restrict *in,
302  char const *restrict in_end, size_t *n_non_newline) {
303  if (ctx->i == 8)
304  ctx->i = 0;
305 
306  if (ctx->i == 0) {
307  char const *t = *in;
308  if (8 <= in_end - *in && memchr(t, '\n', 8) == NULL) {
309  /* This is the common case: no newline. */
310  *in += 8;
311  *n_non_newline = 8;
312  return (char *)t;
313  }
314  }
315 
316  {
317  /* Copy non-newline bytes into BUF. */
318  char const *p = *in;
319  while (p < in_end) {
320  char c = *p++;
321  if (c != '\n') {
322  ctx->buf[ctx->i++] = c;
323  if (ctx->i == 8)
324  break;
325  }
326  }
327 
328  *in = p;
329  *n_non_newline = ctx->i;
330  return ctx->buf;
331  }
332 }

340 /* Decode eight bytes of base32-encoded data, IN, of length INLEN
341  into the output buffer, *OUT, of size *OUTLEN bytes. Return true if
342  decoding is successful, false otherwise. If *OUTLEN is too small,
343  as many bytes as possible are written to *OUT. On return, advance
344  *OUT to point to the byte after the last one written, and decrement
345  *OUTLEN to reflect the number of bytes remaining in *OUT. */
346 static bool decode_8(char const *restrict in, size_t inlen, char *restrict *outp,
347  size_t *outleft) {
348  char *out = *outp;
349  if (inlen < 8)
350  return false;
351 
352  if (!isbase32(in[0]) || !isbase32(in[1]))
353  return false;
354 
355  if (*outleft) {
356  *out++ = ((b32[to_uchar(in[0])] << 3) | (b32[to_uchar(in[1])] >> 2));
357  --*outleft;
358  }
359 
360  if (in[2] == '=') {
361  if (in[3] != '=' || in[4] != '=' || in[5] != '=' || in[6] != '=' ||
362  in[7] != '=')
363  return_false;
364  } else {
365  if (!isbase32(in[2]) || !isbase32(in[3]))
366  return_false;
367 
368  if (*outleft) {
369  *out++ = ((b32[to_uchar(in[1])] << 6) | (b32[to_uchar(in[2])] << 1) |
370  (b32[to_uchar(in[3])] >> 4));
371  --*outleft;
372  }
373 
374  if (in[4] == '=') {
375  if (in[5] != '=' || in[6] != '=' || in[7] != '=')
376  return_false;
377  } else {
378  if (!isbase32(in[4]))
379  return_false;
380 
381  if (*outleft) {
382  *out++ = ((b32[to_uchar(in[3])] << 4) | (b32[to_uchar(in[4])] >> 1));
383  --*outleft;
384  }
385 
386  if (in[5] == '=') {
387  if (in[6] != '=' || in[7] != '=')
388  return_false;
389  } else {
390  if (!isbase32(in[5]) || !isbase32(in[6]))
391  return_false;
392 
393  if (*outleft) {
394  *out++ =
395  ((b32[to_uchar(in[4])] << 7) | (b32[to_uchar(in[5])] << 2) |
396  (b32[to_uchar(in[6])] >> 3));
397  --*outleft;
398  }
399 
400  if (in[7] != '=') {
401  if (!isbase32(in[7]))
402  return_false;
403 
404  if (*outleft) {
405  *out++ =
406  ((b32[to_uchar(in[6])] << 5) | (b32[to_uchar(in[7])]));
407  --*outleft;
408  }
409  }
410  }
411  }
412  }
413 
414  *outp = out;
415  return true;
416 }

418 /* Decode base32-encoded input array IN of length INLEN to output array
419  OUT that can hold *OUTLEN bytes. The input data may be interspersed
420  with newlines. Return true if decoding was successful, i.e. if the
421  input was valid base32 data, false otherwise. If *OUTLEN is too
422  small, as many bytes as possible will be written to OUT. On return,
423  *OUTLEN holds the length of decoded bytes in OUT. Note that as soon
424  as any non-alphabet, non-newline character is encountered, decoding
425  is stopped and false is returned. If INLEN is zero, then process
426  only whatever data is stored in CTX.
427 
428  Initially, CTX must have been initialized via base32_decode_ctx_init.
429  Subsequent calls to this function must reuse whatever state is recorded
430  in that buffer. It is necessary for when a octuple of base32 input
431  bytes spans two input buffers.
432 
433  If CTX is NULL then newlines are treated as garbage and the input
434  buffer is processed as a unit. */
435 
436 bool base32_decode_ctx(struct base32_decode_context *ctx, const char *restrict in,
437  size_t inlen, char *restrict out, size_t *outlen) {
438  size_t outleft = *outlen;
439  bool ignore_newlines = ctx != NULL;
440  bool flush_ctx = false;
441  unsigned int ctx_i = 0;
442 
443  if (ignore_newlines) {
444  ctx_i = ctx->i;
445  flush_ctx = inlen == 0;
446  }
447 
448  while (true) {
449  size_t outleft_save = outleft;
450  if (ctx_i == 0 && !flush_ctx) {
451  while (true) {
452  /* Save a copy of outleft, in case we need to
453  re-parse this block of four bytes. */
454  outleft_save = outleft;
455  if (!decode_8(in, inlen, &out, &outleft))
456  break;
457 
458  in += 8;
459  inlen -= 8;
460  }
461  }
462 
463  if (inlen == 0 && !flush_ctx)
464  break;
465 
466  /* Handle the common case of 72-byte wrapped lines.
467  This also handles any other multiple-of-8-byte wrapping. */
468  if (inlen && *in == '\n' && ignore_newlines) {
469  ++in;
470  --inlen;
471  continue;
472  }
473 
474  /* Restore OUT and OUTLEFT. */
475  out -= outleft_save - outleft;
476  outleft = outleft_save;
477 
478  {
479  char const *in_end = in + inlen;
480  char const *non_nl;
481 
482  if (ignore_newlines)
483  non_nl = get_8(ctx, &in, in_end, &inlen);
484  else
485  non_nl = in; /* Might have nl in this case. */
486 
487  /* If the input is empty or consists solely of newlines
488  (0 non-newlines), then we're done. Likewise if there
489  are fewer than 8 bytes when not
490  flushing context and not treating newlines as
491  garbage. */
492  if (inlen == 0 || (inlen < 8 && !flush_ctx && ignore_newlines)) {
493  inlen = 0;
494  break;
495  }
496  if (!decode_8(non_nl, inlen, &out, &outleft))
497  break;
498 
499  inlen = in_end - in;
500  }
501  }
502 
503  *outlen -= outleft;
504 
505  return inlen == 0;
506 }

508 /* Allocate an output buffer in *OUT, and decode the base32 encoded
509  data stored in IN of size INLEN to the *OUT buffer. On return, the
510  size of the decoded data is stored in *OUTLEN. OUTLEN may be NULL,
511  if the caller is not interested in the decoded length. *OUT may be
512  NULL to indicate an out of memory error, in which case *OUTLEN
513  contains the size of the memory block needed. The function returns
514  true on successful decoding and memory allocation errors. (Use the
515  *OUT and *OUTLEN parameters to differentiate between successful
516  decoding and memory error.) The function returns false if the
517  input was invalid, in which case *OUT is NULL and *OUTLEN is
518  undefined. */
519 bool base32_decode_alloc_ctx(struct base32_decode_context *ctx, const char *in,
520  size_t inlen, char *out, size_t *outlen) {
521  /* This may allocate a few bytes too many, depending on input,
522  but it's not worth the extra CPU time to compute the exact size.
523  The exact size is 5 * inlen / 8, minus one or more bytes if the
524  input is padded with one or more "=".
525  Dividing before multiplying avoids the possibility of overflow. */
526  size_t needlen = 5 * (inlen / 8) + 5;
527 
528  if (needlen > *outlen) {
529  return false;
530  }
531 
532  if (!base32_decode_ctx(ctx, in, inlen, out, &needlen)) {
533  return false;
534  }
535 
536  if (outlen)
537  *outlen = needlen;
538  return true;
539 }
