43 /**
44  * \defgroup CborToJson Converting CBOR to JSON
45  * \brief Group of functions used to convert CBOR to JSON.
46  *
47  * This group contains two functions that can be used to convert a \ref
48  * CborValue object to an equivalent JSON representation. This module attempts
49  * to follow the recommendations from RFC 7049 section 4.1 "Converting from
50  * CBOR to JSON", though it has a few differences. They are noted below.
51  *
52  * These functions produce a "minified" JSON output, with no spacing,
53  * indentation or line breaks. If those are necessary, they need to be applied
54  * in a post-processing phase.
55  *
56  * Note that JSON cannot support all CBOR types with fidelity, so the
57  * conversion is usually lossy. For that reason, TinyCBOR supports adding a set
58  * of metadata JSON values that can be used by a JSON-to-CBOR converter to
59  * restore the original data types.
60  *
61  * The TinyCBOR library does not provide a way to convert from JSON
62  * representation back to encoded form. However, it provides a tool called
63  * \c json2cbor which can be used for that purpose. That tool supports the
64  * metadata format that these functions may produce.
65  *
66  * Either of the functions in this section will attempt to convert exactly one
67  * CborValue object to JSON. Those functions may return any error documented
68  * for the functions for CborParsing. In addition, if the C standard library
69  * stream functions return with error, the text conversion will return with
70  * error CborErrorIO.
71  *
72  * These functions also perform UTF-8 validation in CBOR text strings. If they
73  * encounter a sequence of bytes that is not permitted in UTF-8, they will return
74  * CborErrorInvalidUtf8TextString. That includes encoding of surrogate points
75  * in UTF-8.
76  *
77  * \warning The metadata produced by these functions is not guaranteed to
78  * remain stable. A future update of TinyCBOR may produce different output for
79  * the same input and parsers may be unable to handle it.
80  *
81  * \sa CborParsing, CborPretty, cbor_parser_init()
82  */
83 
84 /**
85  * \addtogroup CborToJson
86  * @{
87  * <h2 class="groupheader">Conversion limitations</h2>
88  *
89  * When converting from CBOR to JSON, there may be information loss. This
90  * section lists the possible scenarios.
91  *
92  * \par Number precision:
93  * ALL JSON numbers, due to its JavaScript heritage, are IEEE 754
94  * double-precision floating point. This means JSON is not capable of
95  * representing all integers numbers outside the range [-(2<sup>53</sup>)+1,
96  * 2<sup>53</sup>-1] and is not capable of representing NaN or infinite. If the
97  * CBOR data contains a number outside the valid range, the conversion will
98  * lose precision. If the input was NaN or infinite, the result of the
99  * conversion will be the JSON null value. In addition, the distinction between
100  * half-, single- and double-precision is lost.
101  *
102  * \par
103  * If enabled, the original value and original type are stored in the metadata.
104  *
105  * \par Non-native types:
106  * CBOR's type system is richer than JSON's, which means some data values
107  * cannot be represented when converted to JSON. The conversion silently turns
108  * them into strings: CBOR simple types become "simple(nn)" where \c nn is the
109  * simple type's value, with the exception of CBOR undefined, which becomes
110  * "undefined", while CBOR byte strings are converted to an Base16, Base64, or
111  * Base64url encoding
112  *
113  * \par
114  * If enabled, the original type is stored in the metadata.
115  *
116  * \par Presence of tags:
117  * JSON has no support for tagged values, so by default tags are dropped when
118  * converting to JSON. However, if the CborConvertObeyByteStringTags option is
119  * active (default), then certain known tags are honored and are used to format
120  * the conversion of the tagged byte string to JSON.
121  *
122  * \par
123  * If the CborConvertTagsToObjects option is active, then the tag and the
124  * tagged value are converted to a JSON object. Otherwise, if enabled, the
125  * last (innermost) tag is stored in the metadata.
126  *
127  * \par Non-string keys in maps:
128  * JSON requires all Object keys to be strings, while CBOR does not. By
129  * default, if a non-string key is found, the conversion fails with error
130  * CborErrorJsonObjectKeyNotString. If the CborConvertStringifyMapKeys option
131  * is active, then the conversion attempts to create a string representation
132  * using CborPretty. Note that the \c json2cbor tool is not able to parse this
133  * back to the original form.
134  *
135  * \par Duplicate keys in maps:
136  * Neither JSON nor CBOR allow duplicated keys, but current TinyCBOR does not
137  * validate that this is the case. If there are duplicated keys in the input,
138  * they will be repeated in the output, which many JSON tools may flag as
139  * invalid. In addition to that, if the CborConvertStringifyMapKeys option is
140  * active, it is possible that a non-string key in a CBOR map will be converted
141  * to a string form that is identical to another key.
142  *
143  * \par
144  * When metadata support is active, the conversion will add extra key-value
145  * pairs to the JSON output so it can store the metadata. It is possible that
146  * the keys for the metadata clash with existing keys in the JSON map.
147  */

173 static CborError dump_bytestring_base16(char **result, CborValue *it) {
174  static const char characters[] = "0123456789abcdef";
175  size_t i;
176  size_t n = 0;
177  uint8_t *buffer;
178  CborError err = cbor_value_calculate_string_length(it, &n);
179  if (err)
180  return err;
181 
182  /* a Base16 (hex) output is twice as big as our buffer */
183  buffer = (uint8_t *)malloc(n * 2 + 1);
184  *result = (char *)buffer;
185 
186  /* let cbor_value_copy_byte_string know we have an extra byte for the terminating
187  * NUL */
188  ++n;
189  err = cbor_value_copy_byte_string(it, buffer + n - 1, &n, it);
190  cbor_assert(err == CborNoError);
191 
192  for (i = 0; i < n; ++i) {
193  uint8_t byte = buffer[n + i];
194  buffer[2 * i] = characters[byte >> 4];
195  buffer[2 * i + 1] = characters[byte & 0xf];
196  }
197  return CborNoError;
198 }

200 static CborError generic_dump_base64(char **result, CborValue *it,
201  const char alphabet[65]) {
202  size_t n = 0, i;
203  uint8_t *buffer, *out, *in;
204  CborError err = cbor_value_calculate_string_length(it, &n);
205  if (err)
206  return err;
207 
208  /* a Base64 output (untruncated) has 4 bytes for every 3 in the input */
209  size_t len = (n + 5) / 3 * 4;
210  out = buffer = (uint8_t *)malloc(len + 1);
211  *result = (char *)buffer;
212 
213  /* we read our byte string at the tail end of the buffer
214  * so we can do an in-place conversion while iterating forwards */
215  in = buffer + len - n;
216 
217  /* let cbor_value_copy_byte_string know we have an extra byte for the terminating
218  * NUL */
219  ++n;
220  err = cbor_value_copy_byte_string(it, in, &n, it);
221  cbor_assert(err == CborNoError);
222 
223  uint_least32_t val = 0;
224  for (i = 0; n - i >= 3; i += 3) {
225  /* read 3 bytes x 8 bits = 24 bits */
226  if (false) {
227 #ifdef __GNUC__
228  } else if (i) {
229  __builtin_memcpy(&val, in + i - 1, sizeof(val));
230  val = cbor_ntohl(val);
231 #endif
232  } else {
233  val = (in[i] << 16) | (in[i + 1] << 8) | in[i + 2];
234  }
235 
236  /* write 4 chars x 6 bits = 24 bits */
237  *out++ = alphabet[(val >> 18) & 0x3f];
238  *out++ = alphabet[(val >> 12) & 0x3f];
239  *out++ = alphabet[(val >> 6) & 0x3f];
240  *out++ = alphabet[val & 0x3f];
241  }
242 
243  /* maybe 1 or 2 bytes left */
244  if (n - i) {
245  /* we can read in[i + 1] even if it's past the end of the string because
246  * we know (by construction) that it's a NUL byte */
247 #ifdef __GNUC__
248  uint16_t val16;
249  __builtin_memcpy(&val16, in + i, sizeof(val16));
250  val = cbor_ntohs(val16);
251 #else
252  val = (in[i] << 8) | in[i + 1];
253 #endif
254  val <<= 8;
255 
256  /* the 65th character in the alphabet is our filler: either '=' or '\0' */
257  out[4] = '\0';
258  out[3] = alphabet[64];
259  if (n - i == 2) {
260  /* write the third char in 3 chars x 6 bits = 18 bits */
261  out[2] = alphabet[(val >> 6) & 0x3f];
262  } else {
263  out[2] = alphabet[64]; /* filler */
264  }
265  out[1] = alphabet[(val >> 12) & 0x3f];
266  out[0] = alphabet[(val >> 18) & 0x3f];
267  } else {
268  out[0] = '\0';
269  }
270 
271  return CborNoError;
272 }

299 static CborError add_value_metadata(FILE *out, CborType type,
300  const ConversionStatus *status) {
301  int flags = status->flags;
302  if (flags & TypeWasTagged) {
303  /* extract the tagged type, which may be JSON native */
304  type = flags & FinalTypeMask;
305  flags &= ~(FinalTypeMask | TypeWasTagged);
306 
307  if (fprintf(out, "\"tag\":\"%" PRIu64 "\"%s", status->lastTag,
308  flags & ~TypeWasTagged ? "," : "") < 0)
309  return CborErrorIO;
310  }
311 
312  if (!flags)
313  return CborNoError;
314 
315  /* print at least the type */
316  if (fprintf(out, "\"t\":%d", type) < 0)
317  return CborErrorIO;
318 
319  if (flags & NumberWasNaN)
320  if (fprintf(out, ",\"v\":\"nan\"") < 0)
321  return CborErrorIO;
322  if (flags & NumberWasInfinite)
323  if (fprintf(out, ",\"v\":\"%sinf\"", flags & NumberWasNegative ? "-" : "") <
324  0)
325  return CborErrorIO;
326  if (flags & NumberPrecisionWasLost)
327  if (fprintf(out, ",\"v\":\"%c%" PRIx64 "\"",
328  flags & NumberWasNegative ? '-' : '+',
329  status->originalNumber) < 0)
330  return CborErrorIO;
331  if (type == CborSimpleType)
332  if (fprintf(out, ",\"v\":%d", (int)status->originalNumber) < 0)
333  return CborErrorIO;
334  return CborNoError;
335 }

337 static CborError find_tagged_type(CborValue *it, CborTag *tag, CborType *type) {
338  CborError err = CborNoError;
339  *type = cbor_value_get_type(it);
340  while (*type == CborTagType) {
341  cbor_value_get_tag(it, tag); /* can't fail */
342  err = cbor_value_advance_fixed(it);
343  if (err)
344  return err;
345 
346  *type = cbor_value_get_type(it);
347  }
348  return err;
349 }

351 static CborError tagged_value_to_json(FILE *out, CborValue *it, int flags,
352  ConversionStatus *status) {
353  CborTag tag;
354  CborError err;
355 
356  if (flags & CborConvertTagsToObjects) {
357  cbor_value_get_tag(it, &tag); /* can't fail */
358  err = cbor_value_advance_fixed(it);
359  if (err)
360  return err;
361 
362  if (fprintf(out, "{\"tag%" PRIu64 "\":", tag) < 0)
363  return CborErrorIO;
364 
365  CborType type = cbor_value_get_type(it);
366  err = value_to_json(out, it, flags, type, status);
367  if (err)
368  return err;
369  if (flags & CborConvertAddMetadata && status->flags) {
370  if (fprintf(out, ",\"tag%" PRIu64 "$cbor\":{", tag) < 0 ||
371  add_value_metadata(out, type, status) != CborNoError ||
372  fputc('}', out) < 0)
373  return CborErrorIO;
374  }
375  if (fputc('}', out) < 0)
376  return CborErrorIO;
377  status->flags = TypeWasNotNative | CborTagType;
378  return CborNoError;
379  }
380 
381  CborType type;
382  err = find_tagged_type(it, &status->lastTag, &type);
383  if (err)
384  return err;
385  tag = status->lastTag;
386 
387  /* special handling of byte strings? */
388  if (type == CborByteStringType &&
389  (flags & CborConvertByteStringsToBase64Url) == 0 &&
390  (tag == CborNegativeBignumTag || tag == CborExpectedBase16Tag ||
391  tag == CborExpectedBase64Tag)) {
392  char *str;
393  const char *pre = "";
394 
395  if (tag == CborNegativeBignumTag) {
396  pre = "~";
397  err = dump_bytestring_base64url(&str, it);
398  } else if (tag == CborExpectedBase64Tag) {
399  err = dump_bytestring_base64(&str, it);
400  } else { /* tag == CborExpectedBase16Tag */
401  err = dump_bytestring_base16(&str, it);
402  }
403  if (err)
404  return err;
405  err = fprintf(out, "\"%s%s\"", pre, str) < 0 ? CborErrorIO : CborNoError;
406  free(str);
407  status->flags = TypeWasNotNative | TypeWasTagged | CborByteStringType;
408  return err;
409  }
410 
411  /* no special handling */
412  err = value_to_json(out, it, flags, type, status);
413  status->flags |= TypeWasTagged | type;
414  return err;
415 }

418  CborType type) {
419  (void)flags; /* unused */
420  (void)type; /* unused */

455 static CborError map_to_json(FILE *out, CborValue *it, int flags,
456  ConversionStatus *status) {
457  const char *comma = "";
458  CborError err;
459  while (!cbor_value_at_end(it)) {
460  char *key;
461  if (fprintf(out, "%s", comma) < 0)
462  return CborErrorIO;
463  comma = ",";
464 
465  CborType keyType = cbor_value_get_type(it);
466  if (likely(keyType == CborTextStringType)) {
467  size_t n = 0;
468  err = cbor_value_dup_text_string(it, &key, &n, it);
469  } else if (flags & CborConvertStringifyMapKeys) {
470  err = stringify_map_key(&key, it, flags, keyType);
471  } else {
472  return CborErrorJsonObjectKeyNotString;
473  }
474  if (err)
475  return err;
476 
477  /* first, print the key */
478  if (fprintf(out, "\"%s\":", key) < 0) {
479  free(key);
480  return CborErrorIO;
481  }
482 
483  /* then, print the value */
484  CborType valueType = cbor_value_get_type(it);
485  err = value_to_json(out, it, flags, valueType, status);
486 
487  /* finally, print any metadata we may have */
488  if (flags & CborConvertAddMetadata) {
489  if (!err && keyType != CborTextStringType) {
490  if (fprintf(out, ",\"%s$keycbordump\":true", key) < 0)
491  err = CborErrorIO;
492  }
493  if (!err && status->flags) {
494  if (fprintf(out, ",\"%s$cbor\":{", key) < 0 ||
495  add_value_metadata(out, valueType, status) != CborNoError ||
496  fputc('}', out) < 0)
497  err = CborErrorIO;
498  }
499  }
500 
501  free(key);
502  if (err)
503  return err;
504  }
505  return CborNoError;
506 }

508 static CborError value_to_json(FILE *out, CborValue *it, int flags, CborType type,
509  ConversionStatus *status) {
510  CborError err;
511  status->flags = 0;
512 
513  switch (type) {
514  case CborArrayType:
515  case CborMapType: {
516  /* recursive type */
517  CborValue recursed;
518  err = cbor_value_enter_container(it, &recursed);
519  if (err) {
520  it->ptr = recursed.ptr;
521  return err; /* parse error */
522  }
523  if (fputc(type == CborArrayType ? '[' : '{', out) < 0)
524  return CborErrorIO;
525 
526  err = (type == CborArrayType)
527  ? array_to_json(out, &recursed, flags, status)
528  : map_to_json(out, &recursed, flags, status);
529  if (err) {
530  it->ptr = recursed.ptr;
531  return err; /* parse error */
532  }
533 
534  if (fputc(type == CborArrayType ? ']' : '}', out) < 0)
535  return CborErrorIO;
536  err = cbor_value_leave_container(it, &recursed);
537  if (err)
538  return err; /* parse error */
539 
540  status->flags = 0; /* reset, there are never conversion errors for us */
541  return CborNoError;
542  }
543 
544  case CborIntegerType: {
545  double num; /* JS numbers are IEEE double precision */
546  uint64_t val;
547  cbor_value_get_raw_integer(it, &val); /* can't fail */
548  num = (double)val;
549 
550  if (cbor_value_is_negative_integer(it)) {
551  num = -num - 1; /* convert to negative */
552  if ((uint64_t)(-num - 1) != val) {
553  status->flags = NumberPrecisionWasLost | NumberWasNegative;
554  status->originalNumber = val;
555  }
556  } else {
557  if ((uint64_t)num != val) {
558  status->flags = NumberPrecisionWasLost;
559  status->originalNumber = val;
560  }
561  }
562  if (fprintf(out, "%.0f", num) <
563  0) /* this number has no fraction, so no decimal points please */
564  return CborErrorIO;
565  break;
566  }
567 
568  case CborByteStringType:
569  case CborTextStringType: {
570  char *str;
571  if (type == CborByteStringType) {
572  err = dump_bytestring_base64url(&str, it);
573  status->flags = TypeWasNotNative;
574  } else {
575  size_t n = 0;
576  err = cbor_value_dup_text_string(it, &str, &n, it);
577  }
578  if (err)
579  return err;
580  err = (fprintf(out, "\"%s\"", str) < 0) ? CborErrorIO : CborNoError;
581  free(str);
582  return err;
583  }
584 
585  case CborTagType:
586  return tagged_value_to_json(out, it, flags, status);
587 
588  case CborSimpleType: {
589  uint8_t simple_type;
590  cbor_value_get_simple_type(it, &simple_type); /* can't fail */
591  status->flags = TypeWasNotNative;
592  status->originalNumber = simple_type;
593  if (fprintf(out, "\"simple(%" PRIu8 ")\"", simple_type) < 0)
594  return CborErrorIO;
595  break;
596  }
597 
598  case CborNullType:
599  if (fprintf(out, "null") < 0)
600  return CborErrorIO;
601  break;
602 
603  case CborUndefinedType:
604  status->flags = TypeWasNotNative;
605  if (fprintf(out, "\"undefined\"") < 0)
606  return CborErrorIO;
607  break;
608 
609  case CborBooleanType: {
610  bool val;
611  cbor_value_get_boolean(it, &val); /* can't fail */
612  if (fprintf(out, val ? "true" : "false") < 0)
613  return CborErrorIO;
614  break;
615  }
616 
617 #ifndef CBOR_NO_FLOATING_POINT
618  case CborDoubleType: {
619  double val;
620  if (false) {
621  float f;
622  case CborFloatType:
623  status->flags = TypeWasNotNative;
624  cbor_value_get_float(it, &f);
625  val = f;
626  } else if (false) {
627  uint16_t f16;
628  case CborHalfFloatType:
629 #ifndef CBOR_NO_HALF_FLOAT_TYPE
630  status->flags = TypeWasNotNative;
