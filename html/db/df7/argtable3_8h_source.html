79 /*
80  * The arg_hdr struct defines properties that are common to all arg_xxx structs.
81  * The argtable library requires each arg_xxx struct to have an arg_hdr
82  * struct as its first data member.
83  * The argtable library functions then use this data to identify the
84  * properties of the command line option, such as its option tags,
85  * datatype string, and glossary strings, and so on.
86  * Moreover, the arg_hdr struct contains pointers to custom functions that
87  * are provided by each arg_xxx struct which perform the tasks of parsing
88  * that particular arg_xxx arguments, performing post-parse checks, and
89  * reporting errors.
90  * These functions are private to the individual arg_xxx source code
91  * and are the pointer to them are initiliased by that arg_xxx struct's
92  * constructor function. The user could alter them after construction
93  * if desired, but the original intention is for them to be set by the
94  * constructor and left unaltered.
95  */
96 struct arg_hdr {
97  char flag; /* Modifier flags: ARG_TERMINATOR, ARG_HASVALUE. */
98  const char *shortopts; /* String defining the short options */
99  const char *longopts; /* String defiing the long options */
100  const char *datatype; /* Description of the argument data type */
101  const char *glossary; /* Description of the option as shown by arg_print_glossary
102  function */
103  int mincount; /* Minimum number of occurences of this option accepted */
104  int maxcount; /* Maximum number of occurences if this option accepted */
105  void *parent; /* Pointer to parent arg_xxx struct */
106  arg_resetfn *resetfn; /* Pointer to parent arg_xxx reset function */
107  arg_scanfn *scanfn; /* Pointer to parent arg_xxx scan function */
108  arg_checkfn *checkfn; /* Pointer to parent arg_xxx check function */
109  arg_errorfn *errorfn; /* Pointer to parent arg_xxx error function */
110  void *priv; /* Pointer to private header data for use by arg_xxx functions */
111 };
