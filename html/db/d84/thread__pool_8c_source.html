108 /* Initialise thread pool */
109 struct thpool_ *thpool_init(int num_threads) {
110 
111  threads_on_hold = 0;
112  threads_keepalive = 1;
113 
114  if (num_threads < 0) {
115  num_threads = 0;
116  }
117 
118  /* Make new thread pool */
119  thpool_ *thpool_p;
120  thpool_p = (struct thpool_ *)malloc(sizeof(struct thpool_));
121  if (thpool_p == NULL) {
122  err("thpool_init(): Could not allocate memory for thread pool\n");
123  return NULL;
124  }
125  thpool_p->num_threads_alive = 0;
126  thpool_p->num_threads_working = 0;
127 
128  /* Initialise the job queue */
129  if (jobqueue_init(&thpool_p->jobqueue) == -1) {
130  err("thpool_init(): Could not allocate memory for job queue\n");
131  free(thpool_p);
132  return NULL;
133  }
134 
135  /* Make threads in pool */
136  thpool_p->threads =
137  (struct thread **)malloc(num_threads * sizeof(struct thread *));
138  if (thpool_p->threads == NULL) {
139  err("thpool_init(): Could not allocate memory for threads\n");
140  jobqueue_destroy(&thpool_p->jobqueue);
141  free(thpool_p);
142  return NULL;
143  }
144 
145  pthread_mutex_init(&(thpool_p->thcount_lock), NULL);
146  pthread_cond_init(&thpool_p->threads_all_idle, NULL);
147 
148  /* Thread init */
149  int n;
150  for (n = 0; n < num_threads; n++) {
151  thread_init(thpool_p, &thpool_p->threads[n], n);
152 #if THPOOL_DEBUG
153  printf("THPOOL_DEBUG: Created thread %d in pool \n", n);
154 #endif
155  }
156 
157  /* Wait for threads to initialize */
158  while (thpool_p->num_threads_alive != num_threads) {
159  }
160 
161  return thpool_p;
162 }

185 void thpool_wait(thpool_ *thpool_p) {
186  pthread_mutex_lock(&thpool_p->thcount_lock);
187  while (thpool_p->jobqueue.len || thpool_p->num_threads_working) {
188  pthread_cond_wait(&thpool_p->threads_all_idle, &thpool_p->thcount_lock);
189  }

193 /* Destroy the threadpool */
194 void thpool_destroy(thpool_ *thpool_p) {
195  /* No need to destory if it's NULL */
196  if (thpool_p == NULL)
197  return;
198 
199  volatile int threads_total = thpool_p->num_threads_alive;
200 
201  /* End each thread 's infinite loop */
202  threads_keepalive = 0;
203 
204  /* Give one second to kill idle threads */
205  double TIMEOUT = 1.0;
206  time_t start, end;
207  double tpassed = 0.0;
208  time(&start);
209  while (tpassed < TIMEOUT && thpool_p->num_threads_alive) {
210  bsem_post_all(thpool_p->jobqueue.has_jobs);
211  time(&end);
212  tpassed = difftime(end, start);
213  }
214 
215  /* Poll remaining threads */
216  while (thpool_p->num_threads_alive) {
217  bsem_post_all(thpool_p->jobqueue.has_jobs);
218  sleep(1);
219  }
220 
221  /* Job queue cleanup */
222  jobqueue_destroy(&thpool_p->jobqueue);
223  /* Deallocs */
224  int n;
225  for (n = 0; n < threads_total; n++) {
226  thread_destroy(thpool_p->threads[n]);
227  }
228  free(thpool_p->threads);
229  free(thpool_p);
230 }

288 /* What each thread is doing
289  *
290  * In principle this is an endless loop. The only time this loop gets interuppted is
291  * once thpool_destroy() is invoked or the program exits.
292  *
293  * @param thread thread that will run this function
294  * @return nothing
295  */
296 static void *thread_do(struct thread *thread_p) {
297 
298  /* Set thread name for profiling and debuging */
299  char thread_name[128] = {0};
300  sprintf(thread_name, "thread-pool-%d", thread_p->id);
301 
302 #if defined(__linux__)
303  /* Use prctl instead to prevent using _GNU_SOURCE flag and implicit declaration
304  */
305  prctl(PR_SET_NAME, thread_name);
306 #elif defined(__APPLE__) && defined(__MACH__)
307  pthread_setname_np(thread_name);
308 #else
309  err("thread_do(): pthread_setname_np is not supported on this system");
310 #endif
311 
312  /* Assure all threads have been created before starting serving */
313  thpool_ *thpool_p = thread_p->thpool_p;
314 
315  /* Register signal handler */
316  struct sigaction act;
317  sigemptyset(&act.sa_mask);
318  act.sa_flags = 0;
319  act.sa_handler = thread_hold;
320  if (sigaction(SIGUSR1, &act, NULL) == -1) {
321  err("thread_do(): cannot handle SIGUSR1");
322  }
323 
324  /* Mark thread as alive (initialized) */
325  pthread_mutex_lock(&thpool_p->thcount_lock);
326  thpool_p->num_threads_alive += 1;
327  pthread_mutex_unlock(&thpool_p->thcount_lock);
328 
329  while (threads_keepalive) {
330 
331  bsem_wait(thpool_p->jobqueue.has_jobs);
332 
333  if (threads_keepalive) {
334 
335  pthread_mutex_lock(&thpool_p->thcount_lock);
336  thpool_p->num_threads_working++;
337  pthread_mutex_unlock(&thpool_p->thcount_lock);
338 
339  /* Read job from queue and execute it */
340  void (*func_buff)(void *);
341  void *arg_buff;
342  job *job_p = jobqueue_pull(&thpool_p->jobqueue);
343  if (job_p) {
344  func_buff = job_p->function;
345  arg_buff = job_p->arg;
346  func_buff(arg_buff);
347  free(job_p);
348  }
349 
350  pthread_mutex_lock(&thpool_p->thcount_lock);
351  thpool_p->num_threads_working--;
352  if (!thpool_p->num_threads_working) {
353  pthread_cond_signal(&thpool_p->threads_all_idle);
354  }
355  pthread_mutex_unlock(&thpool_p->thcount_lock);
356  }
357  }
358  pthread_mutex_lock(&thpool_p->thcount_lock);
359  thpool_p->num_threads_alive--;
360  pthread_mutex_unlock(&thpool_p->thcount_lock);
361 
362  return NULL;
363 }

402 /* Add (allocated) job to queue
403  */
404 static void jobqueue_push(jobqueue *jobqueue_p, struct job *newjob) {
405 
406  pthread_mutex_lock(&jobqueue_p->rwmutex);
407  newjob->prev = NULL;
408 
409  switch (jobqueue_p->len) {
410 
411  case 0: /* if no jobs in queue */
412  jobqueue_p->front = newjob;
413  jobqueue_p->rear = newjob;
414  break;
415 
416  default: /* if jobs in queue */
417  jobqueue_p->rear->prev = newjob;
418  jobqueue_p->rear = newjob;
419  }
420  jobqueue_p->len++;
421 
422  bsem_post(jobqueue_p->has_jobs);
423  pthread_mutex_unlock(&jobqueue_p->rwmutex);
424 }

426 /* Get first job from queue(removes it from queue)
427  * Notice: Caller MUST hold a mutex
428  */
429 static struct job *jobqueue_pull(jobqueue *jobqueue_p) {
430 
431  pthread_mutex_lock(&jobqueue_p->rwmutex);
432  job *job_p = jobqueue_p->front;
433 
434  switch (jobqueue_p->len) {
435 
436  case 0: /* if no jobs in queue */
437  break;
438 
439  case 1: /* if one job in queue */
440  jobqueue_p->front = NULL;
441  jobqueue_p->rear = NULL;
442  jobqueue_p->len = 0;
443  break;
444 
445  default: /* if >1 jobs in queue */
446  jobqueue_p->front = job_p->prev;
447  jobqueue_p->len--;
448  /* more than one job in queue -> post it */
449  bsem_post(jobqueue_p->has_jobs);
450  }
451 
452  pthread_mutex_unlock(&jobqueue_p->rwmutex);
453  return job_p;
454 }
