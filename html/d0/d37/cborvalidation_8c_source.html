52 /**
53  * \enum CborValidationFlags
54  * The CborValidationFlags enum contains flags that control the validation of a
55  * CBOR stream.
56  *
57  * \value CborValidateBasic Validates only the syntactic correctedness of the
58 stream.
59  * \value CborValidateCanonical Validates that the stream is in canonical format,
60 according to
61  * RFC 7049 section 3.9.
62  * \value CborValidateStrictMode Performs strict validation, according to RFC 7049
63 section 3.10.
64  * \value CborValidateStrictest Attempt to perform the strictest validation we
65 know of.
66  *
67  * \value CborValidateShortestIntegrals (Canonical) Validate that integral
68 numbers and lengths are
69  * enconded in their shortest form possible.
70  * \value CborValidateShortestFloatingPoint (Canonical) Validate that floating-point
71 numbers are encoded
72  * in their shortest form possible.
73  * \value CborValidateShortestNumbers (Canonical) Validate both integral and
74 floating-point numbers
75  * are in their shortest form possible.
76  * \value CborValidateNoIndeterminateLength (Canonical) Validate that no string,
77 array or map uses
78  * indeterminate length encoding.
79  * \value CborValidateMapIsSorted (Canonical & Strict mode) Validate that
80 map keys appear in
81  * sorted order.
82  * \value CborValidateMapKeysAreUnique (Strict mode) Validate that map keys are
83 unique.
84  * \value CborValidateTagUse (Strict mode) Validate that known tags
85 are used with the
86  * correct types. This does not validate
87 that the content of
88  * those types is syntactically correct. For
89 example, this
90  * option validates that tag 1
91 (DateTimeString) is used with
92  * a Text String, but it does not validate
93 that the string is
94  * a valid date/time representation.
95  * \value CborValidateUtf8 (Strict mode) Validate that text strings
96 are appropriately
97  * encoded in UTF-8.
98  * \value CborValidateMapKeysAreString Validate that all map keys are text
99 strings.
100  * \value CborValidateNoUndefined Validate that no elements of type
101 "undefined" are present.
102  * \value CborValidateNoTags Validate that no tags are used.
103  * \value CborValidateFiniteFloatingPoint Validate that all floating point numbers
104 are finite (no NaN or
105  * infinities are allowed).
106  * \value CborValidateCompleteData Validate that the stream is complete and
107 there is no more data
108  * in the buffer.
109  * \value CborValidateNoUnknownSimpleTypesSA Validate that all Standards Action
110 simple types are registered
111  * with IANA.
112  * \value CborValidateNoUnknownSimpleTypes Validate that all simple types used are
113 registered with IANA.
114  * \value CborValidateNoUnknownTagsSA Validate that all Standard Actions tags
115 are registered with IANA.
116  * \value CborValidateNoUnknownTagsSR Validate that all Standard Actions and
117 Specification Required tags
118  * are registered with IANA (see below for
119 limitations).
120  * \value CborValidateNoUnkonwnTags Validate that all tags are registered
121 with IANA
122  * (see below for limitations).
123  *
124  * \par Simple type registry
125  * The CBOR specification requires that registration for use of the first 19
126  * simple types must be done by way of Standards Action. The rest of the simple
127  * types only require a specification. The official list can be obtained from
128  * https://www.iana.org/assignments/cbor-simple-values/cbor-simple-values.xhtml.
129  *
130  * \par
131  * There are no registered simple types recognized by this release of TinyCBOR
132  * (beyond those defined by RFC 7049).
133  *
134  * \par Tag registry
135  * The CBOR specification requires that registration for use of the first 23
136  * tags must be done by way of Standards Action. The next up to tag 255 only
137  * require a specification. Finally, all other tags can be registered on a
138  * first-come-first-serve basis. The official list can be ontained from

140  *
141  * \par

300 static inline CborError validate_utf8_string(const void *ptr, size_t n) {
301  const uint8_t *buffer = (const uint8_t *)ptr;
302  const uint8_t *const end = buffer + n;
303  while (buffer < end) {
304  uint32_t uc = get_utf8(&buffer, end);
305  if (uc == ~0U)
306  return CborErrorInvalidUtf8TextString;
307  }
308  return CborNoError;
309 }

311 static inline CborError validate_simple_type(uint8_t simple_type, uint32_t flags) {
312  /* At current time, all known simple types are those from RFC 7049,
313  * which are parsed by the parser into different CBOR types.
314  * That means that if we've got here, the type is unknown */
315  if (simple_type < 32)
316  return (flags & CborValidateNoUnknownSimpleTypesSA)
317  ? CborErrorUnknownSimpleType
318  : CborNoError;
319  return (flags & CborValidateNoUnknownSimpleTypes) ==
320  CborValidateNoUnknownSimpleTypes
321  ? CborErrorUnknownSimpleType
322  : CborNoError;
323 }

325 static inline CborError validate_number(const CborValue *it, CborType type,
326  uint32_t flags) {
327  CborError err = CborNoError;
328  const uint8_t *ptr = it->ptr;
329  size_t bytesUsed, bytesNeeded;
330  uint64_t value;
331 
332  if ((flags & CborValidateShortestIntegrals) == 0)
333  return err;
334  if (type >= CborHalfFloatType && type <= CborDoubleType)
335  return err; /* checked elsewhere */
336 
337  err = _cbor_value_extract_number(&ptr, it->parser->end, &value);
338  if (err)
339  return err;
340 
341  bytesUsed = (size_t)(ptr - it->ptr - 1);
342  bytesNeeded = 0;
343  if (value >= Value8Bit)
344  ++bytesNeeded;
345  if (value > 0xffU)
346  ++bytesNeeded;
347  if (value > 0xffffU)
348  bytesNeeded += 2;
349  if (value > 0xffffffffU)
350  bytesNeeded += 4;
351  if (bytesNeeded < bytesUsed)
352  return CborErrorOverlongEncoding;
353  return CborNoError;
354 }

356 static inline CborError validate_tag(CborValue *it, CborTag tag, uint32_t flags,
357  int recursionLeft) {
358  CborType type = cbor_value_get_type(it);
359  const size_t knownTagCount = sizeof(knownTagData) / sizeof(knownTagData[0]);
360  const struct KnownTagData *tagData = knownTagData;
361  const struct KnownTagData *const knownTagDataEnd = knownTagData + knownTagCount;
362 
363  if (!recursionLeft)
364  return CborErrorNestingTooDeep;
365  if (flags & CborValidateNoTags)
366  return CborErrorExcludedType;
367 
368  /* find the tag data, if any */
369  for (; tagData != knownTagDataEnd; ++tagData) {
370  if (tagData->tag < tag)
371  continue;
372  if (tagData->tag > tag)
373  tagData = NULL;
374  break;
375  }
376  if (tagData == knownTagDataEnd)
377  tagData = NULL;
378 
379  if (flags & CborValidateNoUnknownTags && !tagData) {
380  /* tag not found */
381  if (flags & CborValidateNoUnknownTagsSA && tag < 24)
382  return CborErrorUnknownTag;
383  if ((flags & CborValidateNoUnknownTagsSR) == CborValidateNoUnknownTagsSR &&
384  tag < 256)
385  return CborErrorUnknownTag;
386  if ((flags & CborValidateNoUnknownTags) == CborValidateNoUnknownTags)
387  return CborErrorUnknownTag;
388  }
389 
390  if (flags & CborValidateTagUse && tagData && tagData->types) {
391  uint32_t allowedTypes = tagData->types;
392 
393  /* correct Integer so it's not zero */
394  if (type == CborIntegerType)
395  type = (CborType)(type + 1);
396 
397  while (allowedTypes) {
398  if ((uint8_t)(allowedTypes & 0xff) == type)
399  break;
400  allowedTypes >>= 8;
401  }
402  if (!allowedTypes)
403  return CborErrorInappropriateTagForType;
404  }
405 
406  return validate_value(it, flags, recursionLeft);
407 }

409 #ifndef CBOR_NO_FLOATING_POINT
410 static inline CborError validate_floating_point(CborValue *it, CborType type,
411  uint32_t flags) {
412  CborError err;
413  int r;
414  double val;
415  float valf;
416  uint16_t valf16;
417 
418  if (type != CborDoubleType) {
419  if (type == CborFloatType) {
420  err = cbor_value_get_float(it, &valf);
421  val = valf;
422  } else {
423 #ifdef CBOR_NO_HALF_FLOAT_TYPE
424  (void)valf16;
425  return CborErrorUnsupportedType;
426 #else
427  err = cbor_value_get_half_float(it, &valf16);
428  val = decode_half(valf16);
429 #endif
430  }
431  } else {
432  err = cbor_value_get_double(it, &val);
433  }
434  cbor_assert(err == CborNoError); /* can't fail */
435 
436  r = fpclassify(val);
437  if (r == FP_NAN || r == FP_INFINITE) {
438  if (flags & CborValidateFiniteFloatingPoint)
439  return CborErrorExcludedValue;
440  if (flags & CborValidateShortestFloatingPoint) {
441  if (type == CborDoubleType)
442  return CborErrorOverlongEncoding;
443 #ifndef CBOR_NO_HALF_FLOAT_TYPE
444  if (type == CborFloatType)
445  return CborErrorOverlongEncoding;
446  if (r == FP_NAN && valf16 != 0x7e00)
447  return CborErrorImproperValue;
448  if (r == FP_INFINITE && valf16 != 0x7c00 && valf16 != 0xfc00)
449  return CborErrorImproperValue;
450 #endif
451  }
452  }
453 
454  if (flags & CborValidateShortestFloatingPoint && type > CborHalfFloatType) {
455  if (type == CborDoubleType) {
456  valf = (float)val;
457  if ((double)valf == val)
458  return CborErrorOverlongEncoding;
459  }
460 #ifndef CBOR_NO_HALF_FLOAT_TYPE
461  if (type == CborFloatType) {
462  valf16 = encode_half(valf);
463  if (valf == decode_half(valf16))
464  return CborErrorOverlongEncoding;
465  }
466 #endif
467  }
468 
469  return CborNoError;
470 }
471 #endif

473 static CborError validate_container(CborValue *it, int containerType, uint32_t flags,
474  int recursionLeft) {
475  CborError err;
476  const uint8_t *previous = NULL;
477  const uint8_t *previous_end = NULL;
478 
479  if (!recursionLeft)
480  return CborErrorNestingTooDeep;
481 
482  while (!cbor_value_at_end(it)) {
483  const uint8_t *current = cbor_value_get_next_byte(it);
484 
485  if (containerType == CborMapType) {
486  if (flags & CborValidateMapKeysAreString) {
487  CborType type = cbor_value_get_type(it);
488  if (type == CborTagType) {
489  /* skip the tags */
490  CborValue copy = *it;
491  err = cbor_value_skip_tag(&copy);
492  if (err)
493  return err;
494  type = cbor_value_get_type(&copy);
495  }
496  if (type != CborTextStringType)
497  return CborErrorMapKeyNotString;
498  }
499  }
500 
501  err = validate_value(it, flags, recursionLeft);
502  if (err)
503  return err;
504 
505  if (containerType != CborMapType)
506  continue;
507 
508  if (flags & CborValidateMapIsSorted) {
509  if (previous) {
510  uint64_t len1, len2;
511  const uint8_t *ptr;
512 
513  /* extract the two lengths */
514  ptr = previous;
515  _cbor_value_extract_number(&ptr, it->parser->end, &len1);
516  ptr = current;
517  _cbor_value_extract_number(&ptr, it->parser->end, &len2);
518 
519  if (len1 > len2)
520  return CborErrorMapNotSorted;
521  if (len1 == len2) {
522  size_t bytelen1 = (size_t)(previous_end - previous);
523  size_t bytelen2 = (size_t)(it->ptr - current);
524  int r = memcmp(previous, current,
525  bytelen1 <= bytelen2 ? bytelen1 : bytelen2);
526 
527  if (r == 0 && bytelen1 != bytelen2)
528  r = bytelen1 < bytelen2 ? -1 : +1;
529  if (r > 0)
530  return CborErrorMapNotSorted;
531  if (r == 0 && (flags & CborValidateMapKeysAreUnique) ==
532  CborValidateMapKeysAreUnique)
533  return CborErrorMapKeysNotUnique;
534  }
535  }
536 
537  previous = current;
538  previous_end = it->ptr;
539  }
540 
541  /* map: that was the key, so get the value */
542  err = validate_value(it, flags, recursionLeft);
543  if (err)
544  return err;
545  }
546  return CborNoError;
547 }

549 static CborError validate_value(CborValue *it, uint32_t flags, int recursionLeft) {
550  CborError err;
551  CborType type = cbor_value_get_type(it);
552 
553  if (cbor_value_is_length_known(it)) {
554  err = validate_number(it, type, flags);
555  if (err)
556  return err;
557  } else {
558  if (flags & CborValidateNoIndeterminateLength)
559  return CborErrorUnknownLength;
560  }
561 
562  switch (type) {
563  case CborArrayType:
564  case CborMapType: {
565  /* recursive type */
566  CborValue recursed;
567  err = cbor_value_enter_container(it, &recursed);
568  if (!err)
569  err = validate_container(&recursed, type, flags, recursionLeft - 1);
570  if (err) {
571  it->ptr = recursed.ptr;
572  return err;
573  }
574  err = cbor_value_leave_container(it, &recursed);
575  if (err)
576  return err;
577  return CborNoError;
578  }
579 
580  case CborIntegerType: {
581  uint64_t val;
582  err = cbor_value_get_raw_integer(it, &val);
583  cbor_assert(err == CborNoError); /* can't fail */
584 
585  break;
586  }
587 
588  case CborByteStringType:
589  case CborTextStringType: {
590  size_t n = 0;
591  const void *ptr;
592 
593  err = _cbor_value_prepare_string_iteration(it);
594  if (err)
595  return err;
596 
597  while (1) {
598  CborValue next;
599  err = _cbor_value_get_string_chunk(it, &ptr, &n, &next);
600  if (err)
601  return err;
602  if (ptr) {
603  err = validate_number(it, type, flags);
604  if (err)
605  return err;
606  }
607 
608  *it = next;
609  if (!ptr)
610  break;
611 
612  if (type == CborTextStringType && flags & CborValidateUtf8) {
613  err = validate_utf8_string(ptr, n);
614  if (err)
615  return err;
616  }
617  }
618 
619  return CborNoError;
620  }
621 
622  case CborTagType: {
623  CborTag tag;
624  err = cbor_value_get_tag(it, &tag);
625  cbor_assert(err == CborNoError); /* can't fail */
626 
627  err = cbor_value_advance_fixed(it);
628  if (err)
629  return err;
630  err = validate_tag(it, tag, flags, recursionLeft - 1);
631  if (err)
632  return err;
633 
634  return CborNoError;
635  }
636 
637  case CborSimpleType: {
638  uint8_t simple_type;
639  err = cbor_value_get_simple_type(it, &simple_type);
640  cbor_assert(err == CborNoError); /* can't fail */
641  err = validate_simple_type(simple_type, flags);
642  if (err)
643  return err;
644  break;
645  }
646 
647  case CborNullType:
648  case CborBooleanType:
649  break;
650 
651  case CborUndefinedType:
652  if (flags & CborValidateNoUndefined)
653  return CborErrorExcludedType;
654  break;
655 
656  case CborHalfFloatType:
657  case CborFloatType:
658  case CborDoubleType: {
659 #ifdef CBOR_NO_FLOATING_POINT
660  return CborErrorUnsupportedType;
661 #else
662  err = validate_floating_point(it, type, flags);
663  if (err)
664  return err;
665  break;
666 #endif /* !CBOR_NO_FLOATING_POINT */
667  }
668 
669  case CborInvalidType:
670  return CborErrorUnknownType;
671  }
672 
673  err = cbor_value_advance_fixed(it);
674  return err;
675 }

677 /**
