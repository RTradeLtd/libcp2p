36 /*! @brief returns a new socket client connected to `addr:port`
37  * @param thl an instance of a thread logger
38  * @param addr the multiaddr to connect to
39  */
40 socket_client_t *new_socket_client(thread_logger *thl, multi_addr_t *addr) {
41  char ip[1024];
42 
43  int rc = multiaddress_get_ip_address(addr, ip);
44  if (rc != 1) {
45  return NULL;
46  }
47  int port = multiaddress_get_ip_port(addr);
48  if (port == -1) {
49  return NULL;
50  }
51  char cport[7];
52  sprintf(cport, "%i", port);
53 
54  addr_info hints;
55  memset(&hints, 0, sizeof(hints));
56  bool is_tcp = false;
57  bool is_udp = false;
58 
59  if (strstr(addr->string, "/tcp/") != NULL) {
60  is_tcp = true;
61  hints.ai_family = AF_INET;
62  hints.ai_socktype = SOCK_STREAM;
63  hints.ai_flags = AI_PASSIVE;
64  }
65  if (strstr(addr->string, "/udp/") != NULL) {
66  is_udp = true;
67  hints.ai_family = AF_INET;
68  hints.ai_socktype = SOCK_DGRAM;
69  hints.ai_flags = AI_PASSIVE;
70  }
71 
72  if (is_tcp == false && is_udp == false) {
73  return NULL;
74  }
75 
76  addr_info *peer_address;
77  rc = getaddrinfo(ip, cport, &hints, &peer_address);
78  if (rc != 0) {
79  freeaddrinfo(peer_address);
80  return NULL;
81  }
82 
83  int client_socket_num = get_new_socket(thl, peer_address, NULL, 0, true);
84  if (client_socket_num == -1) {
85  thl->log(thl, 0, "failed to get_new_socket", LOG_LEVELS_ERROR);
86  freeaddrinfo(peer_address);
87  return NULL;
88  }
89 
90  socket_client_t *sock_client =
91  calloc(sizeof(sock_client), sizeof(sock_client) + sizeof(peer_address));
92  if (sock_client == NULL) {
93  thl->log(thl, 0, "failed to calloc socket_client_t", LOG_LEVELS_ERROR);
94  return NULL;
95  }
96  sock_client->socket_number = client_socket_num;
97  sock_client->peer_address = peer_address;
98 
99  thl->log(thl, 0, "client successfully created", LOG_LEVELS_INFO);
100 
101  return sock_client;
102 }
