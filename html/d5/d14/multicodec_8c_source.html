487 /**
488  * Encode a char array using the specified encoding
489  * @param codecName a standard name of the codec
490  * @param inData the incoming data to be encoded
491  * @param inDataLength the length of the inData
492  * @param outData a place to store the outgoing data that is at least
493  * maxOutDataLength in size
494  * @param maxOutDataLength the max size for the output
495  * @returns Success: pointer to an instance of multicodec_encoded_t
496  * @returns Failure: NULL pointer
497  */
498 multicodec_encoded_t *multicodec_encode(char *codec, char *inData,
499  size_t inDataLength, char *outData,
500  size_t maxOutDataLength) {
501 
502  // make sure the codec is supported
503  if (multicodec_is_valid(codec) == false) {
504  return NULL;
505  }
506 
507  multicodec_encoded_t *encoded = calloc(1, sizeof(multicodec_encoded_t));
508  if (encoded == NULL) {
509  return NULL;
510  }
511  memcpy(
512  outData, codec,
513  strlen(codec) // these is no guarantee tehse include a null terminating zero
514  );
515  memcpy(outData + strlen(codec), // copy inData to outData offset my strlen(codec)
516  inData, inDataLength);
517  encoded->codec = codec;
518  encoded->data = outData;
519  return encoded;
520 }

522 /**
523  * Decode a multicodec encoded data array
524  * @param encoded an instance of multicodec_encoded_t holding the encoded data and
525  * codec used
526  * @param outData the array to put the results in
527  * @param maxOutDataLength the maximum size of the data
528  * @returns an error code or 0 if all is well
529  */
530 int multicodec_decode(multicodec_encoded_t *encoded, char *outData,
531  size_t maxOutDataLength) {
532 
533  // make sure the codec is supported
534  if (multicodec_is_valid(encoded->codec) == false) {
535  return 1;
536  }
537 
538  memcpy(outData,
539  /*!
540  * @note copy from inData starting at offset, do `+1` to account for array
541  * index starting at 0,
542  * @note and thus the next I value is the start of the actual data,
543  * whereas if we simply did
544  * @note `inData + i` we would be starting at the end of the codec part
545  */
546  encoded->data + strlen(encoded->codec), maxOutDataLength);
547  return 0;
548 }

550 /*!
551  * @brief used to check if the provided codec is a valid codec
552  * @param codec the codec value to check
553  * @returns Success: true
554  * @returns Failure: false
555  */
556 bool multicodec_is_valid(char *codec) {
557  for (int i = 0; i < CODECS_COUNT; i++) {
558  if (strcmp(codec, CODECS_LIST[i]) == 0) {
559  return true;
560  }
561  }
562  return false;
