41 /**
42  * \defgroup CborParsing Parsing CBOR streams
43  * \brief Group of functions used to parse CBOR streams.
44  *
45  * TinyCBOR provides functions for pull-based stream parsing of a CBOR-encoded
46  * payload. The main data type for the parsing is a CborValue, which behaves
47  * like an iterator and can be used to extract the encoded data. It is first
48  * initialized with a call to cbor_parser_init() and is usually used to extract
49  * exactly one item, most often an array or map.
50  *
51  * Nested CborValue objects can be parsed using cbor_value_enter_container().
52  * Each call to cbor_value_enter_container() must be matched by a call to
53  * cbor_value_leave_container(), with the exact same parameters.
54  *
55  * The example below initializes a CborParser object, begins the parsing with a
56  * CborValue and decodes a single integer:
57  *
58  * \code
59  * int extract_int(const uint8_t *buffer, size_t len)
60  * {
61  * CborParser parser;
62  * CborValue value;
63  * int result;
64  * cbor_parser_init(buffer, len, 0, &parser, &value);
65  * cbor_value_get_int(&value, &result);
66  * return result;
67  * }
68  * \endcode
69  *
70  * The code above does no error checking, which means it assumes the data comes
71  * from a source trusted to send one properly-encoded integer. The following
72  * example does the exact same operation, but includes error checking and
73  * returns 0 on parsing failure:
74  *
75  * \code
76  * int extract_int(const uint8_t *buffer, size_t len)
77  * {
78  * CborParser parser;
79  * CborValue value;
80  * int result;
81  * if (cbor_parser_init(buffer, len, 0, &parser, &value) != CborNoError)
82  * return 0;
83  * if (!cbor_value_is_integer(&value) ||
84  * cbor_value_get_int(&value, &result) != CborNoError)
85  * return 0;
86  * return result;
87  * }
88  * \endcode
89  *
90  * Note, in the example above, that one can't distinguish a parsing failure
91  * from an encoded value of zero. Reporting a parsing error is left as an
92  * exercise to the reader.
93  *
94  * The code above does not execute a range-check either: it is possible that
95  * the value decoded from the CBOR stream encodes a number larger than what can
96  * be represented in a variable of type \c{int}. If detecting that case is
97  * important, the code should call cbor_value_get_int_checked() instead.
98  *
99  * <h3 class="groupheader">Memory and parsing constraints</h3>
100  *
101  * TinyCBOR is designed to run with little memory and with minimal overhead.
102  * Except where otherwise noted, the parser functions always run on constant
103  * time (O(1)), do not recurse and never allocate memory (thus, stack usage is
104  * bounded and is O(1)).
105  *
106  * <h3 class="groupheader">Error handling and preconditions</h3>
107  *
108  * All functions operating on a CborValue return a CborError condition, with
109  * CborNoError standing for the normal situation in which no parsing error
110  * occurred. All functions may return parsing errors in case the stream cannot
111  * be decoded properly, be it due to corrupted data or due to reaching the end
112  * of the input buffer.
113  *
114  * Error conditions must not be ignored. All decoder functions have undefined
115  * behavior if called after an error has been reported, and may crash.
116  *
117  * Some functions are also documented to have preconditions, like
118  * cbor_value_get_int() requiring that the input be an integral value.
119  * Violation of preconditions also results in undefined behavior and the
120  * program may crash.
121  */

162 CborError CBOR_INTERNAL_API_CC _cbor_value_extract_number(const uint8_t **ptr,
163  const uint8_t *end,
164  uint64_t *len) {
165  size_t bytesNeeded;
166  uint8_t additional_information = **ptr & SmallValueMask;
167  ++*ptr;
168  if (additional_information < Value8Bit) {
169  *len = additional_information;
170  return CborNoError;
171  }
172  if (unlikely(additional_information > Value64Bit))
173  return CborErrorIllegalNumber;
174 
175  bytesNeeded = (size_t)(1 << (additional_information - Value8Bit));
176  if (unlikely(bytesNeeded > (size_t)(end - *ptr))) {
177  return CborErrorUnexpectedEOF;
178  } else if (bytesNeeded == 1) {
179  *len = (uint8_t)(*ptr)[0];
180  } else if (bytesNeeded == 2) {
181  *len = get16(*ptr);
182  } else if (bytesNeeded == 4) {
183  *len = get32(*ptr);
184  } else {
185  *len = get64(*ptr);
186  }
187  *ptr += bytesNeeded;
188  return CborNoError;
189 }

211 static CborError preparse_value(CborValue *it) {
212  enum {
213  /* flags to keep */
214  FlagsToKeep = CborIteratorFlag_ContainerIsMap | CborIteratorFlag_NextIsMapKey
215  };
216  const CborParser *parser = it->parser;
217  it->type = CborInvalidType;
218 
219  /* are we at the end? */
220  if (it->ptr == parser->end)
221  return CborErrorUnexpectedEOF;
222 
223  uint8_t descriptor = *it->ptr;
224  uint8_t type = descriptor & MajorTypeMask;
225  it->type = type;
226  it->flags &= FlagsToKeep;
227  it->extra = (descriptor &= SmallValueMask);
228 
229  if (descriptor > Value64Bit) {
230  if (unlikely(descriptor != IndefiniteLength))
231  return type == CborSimpleType ? CborErrorUnknownType
232  : CborErrorIllegalNumber;
233  if (likely(!is_fixed_type(type))) {
234  /* special case */
235  it->flags |= CborIteratorFlag_UnknownLength;
236  it->type = type;
237  return CborNoError;
238  }
239  return type == CborSimpleType ? CborErrorUnexpectedBreak
240  : CborErrorIllegalNumber;
241  }
242 
243  size_t bytesNeeded =
244  descriptor < Value8Bit ? 0 : (1 << (descriptor - Value8Bit));
245  if (bytesNeeded + 1 > (size_t)(parser->end - it->ptr))
246  return CborErrorUnexpectedEOF;
247 
248  uint8_t majortype = type >> MajorTypeShift;
249  if (majortype == NegativeIntegerType) {
250  it->flags |= CborIteratorFlag_NegativeInteger;
251  it->type = CborIntegerType;
252  } else if (majortype == SimpleTypesType) {
253  switch (descriptor) {
254  case FalseValue:
255  it->extra = false;
256  it->type = CborBooleanType;
257  break;
258 
259  case SinglePrecisionFloat:
260  case DoublePrecisionFloat:
261  it->flags |= CborIteratorFlag_IntegerValueTooLarge;
262  /* fall through */
263  case TrueValue:
264  case NullValue:
265  case UndefinedValue:
266  case HalfPrecisionFloat:
267  it->type = *it->ptr;
268  break;
269 
270  case SimpleTypeInNextByte:
271  it->extra = (uint8_t)it->ptr[1];
272 #ifndef CBOR_PARSER_NO_STRICT_CHECKS
273  if (unlikely(it->extra < 32)) {
274  it->type = CborInvalidType;
275  return CborErrorIllegalSimpleType;
276  }
277 #endif
278  break;
279 
280  case 28:
281  case 29:
282  case 30:
283  case Break:
284  cbor_assert(false); /* these conditions can't be reached */
285  return CborErrorUnexpectedBreak;
286  }
287  return CborNoError;
288  }
289 
290  /* try to decode up to 16 bits */
291  if (descriptor < Value8Bit)
292  return CborNoError;
293 
294  if (descriptor == Value8Bit)
295  it->extra = (uint8_t)it->ptr[1];
296  else if (descriptor == Value16Bit)
297  it->extra = get16(it->ptr + 1);
298  else
299  it->flags |=
300  CborIteratorFlag_IntegerValueTooLarge; /* Value32Bit or Value64Bit */
301  return CborNoError;
302 }

304 static CborError preparse_next_value_nodecrement(CborValue *it) {
305  if (it->remaining == UINT32_MAX && it->ptr != it->parser->end &&
306  *it->ptr == (uint8_t)BreakByte) {
307  /* end of map or array */
308  if ((it->flags & CborIteratorFlag_ContainerIsMap &&
309  it->flags & CborIteratorFlag_NextIsMapKey) ||
310  it->type == CborTagType) {
311  /* but we weren't expecting it! */
312  return CborErrorUnexpectedBreak;
313  }
314  ++it->ptr;
315  it->type = CborInvalidType;
316  it->remaining = 0;
317  return CborNoError;
318  }
319 
320  return preparse_value(it);
321 }

323 static CborError preparse_next_value(CborValue *it) {
324  /* tags don't count towards item totals or whether we've successfully
325  * read a map's key or value */
326  bool itemCounts = it->type != CborTagType;
327 
328  if (it->remaining != UINT32_MAX) {
329  if (itemCounts && --it->remaining == 0) {
330  it->type = CborInvalidType;
331  return CborNoError;
332  }
333  }
334  if (itemCounts) {
335  /* toggle the flag indicating whether this was a map key */
336  it->flags ^= CborIteratorFlag_NextIsMapKey;
337  }
338  return preparse_next_value_nodecrement(it);
339 }

341 static CborError advance_internal(CborValue *it) {
342  uint64_t length;
343  CborError err = _cbor_value_extract_number(&it->ptr, it->parser->end, &length);
344  cbor_assert(err == CborNoError);
345 
346  if (it->type == CborByteStringType || it->type == CborTextStringType) {
347  cbor_assert(length == (size_t)length);
348  cbor_assert((it->flags & CborIteratorFlag_UnknownLength) == 0);
349  it->ptr += length;
350  }
351 
352  return preparse_next_value(it);

355 /** \internal
356  *
357  * Decodes the CBOR integer value when it is larger than the 16 bits available
358  * in value->extra. This function requires that value->flags have the
359  * CborIteratorFlag_IntegerValueTooLarge flag set.
360  *
361  * This function is also used to extract single- and double-precision floating
362  * point values (SinglePrecisionFloat == Value32Bit and DoublePrecisionFloat ==
363  * Value64Bit).
364  */
365 uint64_t _cbor_value_decode_int64_internal(const CborValue *value) {
366  cbor_assert(value->flags & CborIteratorFlag_IntegerValueTooLarge ||
367  value->type == CborFloatType || value->type == CborDoubleType);
368 
369  /* since the additional information can only be Value32Bit or Value64Bit,
370  * we just need to test for the one bit those two options differ */
371  cbor_assert((*value->ptr & SmallValueMask) == Value32Bit ||
372  (*value->ptr & SmallValueMask) == Value64Bit);
373  if ((*value->ptr & 1) == (Value32Bit & 1))
374  return get32(value->ptr + 1);
375 
376  cbor_assert((*value->ptr & SmallValueMask) == Value64Bit);
377  return get64(value->ptr + 1);
378 }

495 static CborError advance_recursive(CborValue *it, int nestingLevel) {
496  CborError err;
497  CborValue recursed;
498 
499  if (is_fixed_type(it->type))
500  return advance_internal(it);
501 
502  if (!cbor_value_is_container(it)) {
503  size_t len = SIZE_MAX;
504  return _cbor_value_copy_string(it, NULL, &len, it);
505  }
506 
507  /* map or array */
508  if (nestingLevel == 0)
509  return CborErrorNestingTooDeep;
510 
511  err = cbor_value_enter_container(it, &recursed);
512  if (err)
513  return err;
514  while (!cbor_value_at_end(&recursed)) {
515  err = advance_recursive(&recursed, nestingLevel - 1);
516  if (err)
517  return err;
518  }
519  return cbor_value_leave_container(it, &recursed);
520 }

524  * Unlike cbor_value_advance_fixed(), this function can be called on a CBOR
525  * value of any type. However, if the type is a container (map or array) or a
526  * string with a chunked payload, this function will not run in constant time

540  return CborErrorAdvancePastEOF;

588 /**
589  * Creates a CborValue iterator pointing to the first element of the container
590  * represented by \a it and saves it in \a recursed. The \a it container object
591  * needs to be kept and passed again to cbor_value_leave_container() in order
592  * to continue iterating past this container.
593  *
594  * The \a it CborValue iterator must point to a container.
595  *
596  * \sa cbor_value_is_container(), cbor_value_leave_container(), cbor_value_advance()
597  */
598 CborError cbor_value_enter_container(const CborValue *it, CborValue *recursed) {
599  cbor_static_assert(CborIteratorFlag_ContainerIsMap ==
600  (CborMapType & ~CborArrayType));
601  cbor_assert(cbor_value_is_container(it));
602  *recursed = *it;
603 
604  if (it->flags & CborIteratorFlag_UnknownLength) {
605  recursed->remaining = UINT32_MAX;
606  ++recursed->ptr;
607  } else {
608  uint64_t len;
609  CborError err =
610  _cbor_value_extract_number(&recursed->ptr, recursed->parser->end, &len);
611  cbor_assert(err == CborNoError);
612 
613  recursed->remaining = (uint32_t)len;
614  if (recursed->remaining != len || len == UINT32_MAX) {
615  /* back track the pointer to indicate where the error occurred */
616  recursed->ptr = it->ptr;
617  return CborErrorDataTooLarge;
618  }
619  if (recursed->type == CborMapType) {
620  /* maps have keys and values, so we need to multiply by 2 */
621  if (recursed->remaining > UINT32_MAX / 2) {
622  /* back track the pointer to indicate where the error occurred */
623  recursed->ptr = it->ptr;
624  return CborErrorDataTooLarge;
625  }
626  recursed->remaining *= 2;

820  * result. If the iterator \a value does not point to an integer value, the
821  * behavior is undefined, so checking with \ref cbor_value_get_type or with
822  * \ref cbor_value_is_integer is recommended.
823  *
824  * Unlike \ref cbor_value_get_int64(), this function performs a check to see if the
825  * stored integer fits in \a result without data loss. If the number is outside
826  * the valid range for the data type, this function returns the recoverable
827  * error CborErrorDataTooLarge. In that case, use either
828  * cbor_value_get_uint64() (if the number is positive) or
829  * cbor_value_get_raw_integer().
830  *
831  * \sa cbor_value_get_type(), cbor_value_is_valid(), cbor_value_is_integer(),
832  * cbor_value_get_int64()

835  uint64_t v;
836  cbor_assert(cbor_value_is_integer(value));
837  v = _cbor_value_extract_int64_helper(value);
838 
839  /* Check before converting, as the standard says (C11 6.3.1.3 paragraph 3):
840  * "[if] the new type is signed and the value cannot be represented in it; either
841  * the result is implementation-defined or an implementation-defined signal is
842  * raised."
843  *
844  * The range for int64_t is -2^63 to 2^63-1 (int64_t is required to be
845  * two's complement, C11 7.20.1.1 paragraph 3), which in CBOR is
846  * represented the same way, differing only on the "sign bit" (the major
847  * type).
848  */
849 
850  if (unlikely(v > (uint64_t)INT64_MAX))
851  return CborErrorDataTooLarge;
852 
853  *result = v;
854  if (value->flags & CborIteratorFlag_NegativeInteger)
855  *result = -*result - 1;
856  return CborNoError;
857 }

1013 static CborError get_string_chunk(CborValue *it, const void **bufferptr,
1014  size_t *len) {
1015  CborError err;
1016 
1017  /* Possible states:
1018  * length known | iterating | meaning
1019  * no | no | before the first chunk of a chunked string
1020  * yes | no | at a non-chunked string
1021  * no | yes | second or later chunk
1022  * yes | yes | after a non-chunked string
1023  */
1024  if (it->flags & CborIteratorFlag_IteratingStringChunks) {
1025  /* already iterating */
1026  if (cbor_value_is_length_known(it)) {
1027  /* if the length was known, it wasn't chunked, so finish iteration */
1028  goto last_chunk;
1029  }
1030  } else {
1031  prepare_string_iteration(it);
1032  }
1033 
1034  /* are we at the end? */
1035  if (it->ptr == it->parser->end)
1036  return CborErrorUnexpectedEOF;
1037 
1038  if (*it->ptr == BreakByte) {
1039  /* last chunk */
1040  ++it->ptr;
1041  last_chunk:
1042  *bufferptr = NULL;
1043  *len = 0;
1044  return preparse_next_value(it);
1045  } else if ((uint8_t)(*it->ptr & MajorTypeMask) == it->type) {
1046  err = extract_length(it->parser, &it->ptr, len);
1047  if (err)
1048  return err;
1049  if (*len > (size_t)(it->parser->end - it->ptr))
1050  return CborErrorUnexpectedEOF;
1051 
1052  *bufferptr = it->ptr;
1053  it->ptr += *len;
1054  } else {
1055  return CborErrorIllegalType;
1056  }
1057 
1058  it->flags |= CborIteratorFlag_IteratingStringChunks;
1059  return CborNoError;
1060 }

1094 static CborError iterate_string_chunks(const CborValue *value, char *buffer,
1095  size_t *buflen, bool *result, CborValue *next,
1096  IterateFunction func) {
1097  CborError err;
1098  CborValue tmp;
1099  size_t total = 0;
1100  const void *ptr;
1101 
1102  cbor_assert(cbor_value_is_byte_string(value) ||
1103  cbor_value_is_text_string(value));
1104  if (!next)
1105  next = &tmp;
1106  *next = *value;
1107  *result = true;
1108 
1109  while (1) {
1110  size_t newTotal;
1111  size_t chunkLen;
1112  err = get_string_chunk(next, &ptr, &chunkLen);
1113  if (err)
1114  return err;
1115  if (!ptr)
1116  break;
1117 
1118  if (unlikely(add_check_overflow(total, chunkLen, &newTotal)))
1119  return CborErrorDataTooLarge;
1120 
1121  if (*result && *buflen >= newTotal)
1122  *result = !!func(buffer + total, (const uint8_t *)ptr, chunkLen);
1123  else
1124  *result = false;
1125 
1126  total = newTotal;
1127  }
1128 
1129  /* is there enough room for the ending NUL byte? */
1130  if (*result && *buflen > total) {
1131  uint8_t nul[] = {0};
1132  *result = !!func(buffer + total, nul, 1);
1133  }
1134  *buflen = total;
1135  return CborNoError;
1136 }
