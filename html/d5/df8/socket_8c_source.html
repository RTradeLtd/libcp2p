53 /*! @brief gets an available socket attached to bind_address
54  * @return Success: file descriptor socket number greater than 0
55  * @return Failure: -1
56  * initializers a socket attached to bind_address with sock_opts, and binds the
57  * address
58  */
59 int get_new_socket(thread_logger *thl, addr_info *bind_address,
60  SOCKET_OPTS sock_opts[], int num_opts, bool is_client,
61  bool is_tcp) {
62  // creates the socket and gets us its file descriptor
63  int listen_socket_num =
64  socket(bind_address->ai_family, bind_address->ai_socktype,
65  bind_address->ai_protocol);
66  if (listen_socket_num <= 0) {
67  LOG_ERROR(thl, 0, "socket creation failed");
68  return -1;
69  }
70  int one;
71  int rc;
72  bool passed;
73  for (int i = 0; i < num_opts; i++) {
74  switch (sock_opts[i]) {
75  case REUSEADDR:
76  one = 1;
77  // set socket options before doing anything else
78  // i tried setting it after listen, but I don't think that works
79  rc = setsockopt(listen_socket_num, SOL_SOCKET, SO_REUSEADDR, &one,
80  sizeof(int));
81  if (rc != 0) {
82  LOG_ERROR(thl, 0, "failed to set socket reuse addr");
83  return -1;
84  }
85  LOG_INFO(thl, 0, "set socket opt REUSEADDR");
86  break;
87  case BLOCK:
88  passed = set_socket_blocking_status(listen_socket_num, true);
89  if (passed == false) {
90  LOG_ERROR(thl, 0, "failed to set socket blocking mode");
91  return -1;
92  }
93  LOG_INFO(thl, 0, "set socket opt BLOCK");
94  break;
95  case NOBLOCK:
96  passed = set_socket_blocking_status(listen_socket_num, false);
97  if (passed == false) {
98  LOG_ERROR(thl, 0, "failed to set socket blocking mode");
99  return -1;
100  }
101  LOG_INFO(thl, 0, "set socket opt NOBLOCK");
102  break;
103  default:
104  LOG_ERROR(thl, 0, "invalid socket option");
105  return -1;
106  }
107  }
108 
109  // if client skip bind
110  if (is_client == true) {
111  if (is_tcp) {
112  /*! @todo should we not do this on UDP connections?? */
113  rc = connect(listen_socket_num, bind_address->ai_addr,
114  bind_address->ai_addrlen);
115  if (rc != 0) {
116  close(listen_socket_num);
117  return -1;
118  }
119  }
120  return listen_socket_num;
121  }
122 
123  // binds the address to the socket
124  bind(listen_socket_num, bind_address->ai_addr, bind_address->ai_addrlen);
125  if (errno != 0) {
126  LOGF_ERROR(thl, 0, "socket bind failed with error %s", strerror(errno));
127  return -1;
128  }
129  return listen_socket_num;
130 }

132 /*! @brief used to enable/disable blocking sockets
133  * @return Failure: false
134  * @return Success: true
135  * @note see
136  * https://stackoverflow.com/questions/1543466/how-do-i-change-a-tcp-socket-to-be-non-blocking/1549344#1549344
137  */
138 bool set_socket_blocking_status(int fd, bool blocking) {
139  if (fd < 0) {
140  return false;
141  } else {
142  int flags = fcntl(fd, F_GETFL, 0);
143  if (flags == -1) {
144  return false;
145  }
146  flags = blocking ? (flags & ~O_NONBLOCK) : (flags | O_NONBLOCK);
147  return (fcntl(fd, F_SETFL, flags) == 0) ? true : false;
148  }
149 }

209 /*!
210  * @brief returns an addr_info representation of the multiaddress
211  * @details useful for taking a multi address and getting the needed information for
212  * using
213  * @details the address with the sendto function
214  * @param address the multi address to parse
215  * @note does not free up resources associated with address param
216  * @warning only supports TCP and UDP multiaddress(es)
217  * @return Success: pointer to an addr_info instance
218  * @return Failure: NULL pointer
219  */
220 addr_info *multi_addr_to_addr_info(multi_addr_t *address) {
221  bool is_udp = false;
222  bool is_tcp = false;
223  addr_info hints;
224  memset(&hints, 0, sizeof(addr_info));
225 
226  if (strstr(address->string, "/tcp/") != NULL) {
227  hints.ai_socktype = SOCK_STREAM;
228  is_tcp = true;
229  }
230 
231  if (strstr(address->string, "/udp/") != NULL) {
232  hints.ai_socktype = SOCK_DGRAM;
233  is_udp = true;
234  }
235 
236  if (is_udp == false && is_tcp == false) {
237  return NULL;
238  }
239 
240  char ip[1024];
241  char cport[7];
242  memset(ip, 0, 1024);
243  memset(cport, 0, 1024);
244 
245  int rc = multi_address_get_ip_address(address, ip);
246  if (rc != 1) {
247  return NULL;
248  }
249 
250  int port = multi_address_get_ip_port(address);
251  if (port == -1) {
252  return NULL;
253  }
254  sprintf(cport, "%i", port);
255 
256  int ip_family = multi_address_get_ip_family(address);
257  if (ip_family == 0) {
258  return NULL;
259  }
260 
261  hints.ai_family = ip_family;
262  hints.ai_flags = AI_PASSIVE;
263 
264  addr_info *ret_address;
265 
266  rc = getaddrinfo(ip, cport, &hints, &ret_address);
267  if (rc != 0) {
268  freeaddrinfo(ret_address);
269  return NULL;
270  }
271 
272  return ret_address;
273 }
