40  * \fn CborError cbor_value_dup_text_string(const CborValue *value, char **buffer,
41  * size_t *buflen, CborValue *next)
42  *
43  * Allocates memory for the string pointed by \a value and copies it into this
44  * buffer. The pointer to the buffer is stored in \a buffer and the number of
45  * bytes copied is stored in \a buflen (those variables must not be NULL).
46  *
47  * If the iterator \a value does not point to a text string, the behaviour is
48  * undefined, so checking with \ref cbor_value_get_type or \ref
49  * cbor_value_is_text_string is recommended.
50  *
51  * If \c malloc returns a NULL pointer, this function will return error
52  * condition \ref CborErrorOutOfMemory.
53  *
54  * On success, \c{*buffer} will contain a valid pointer that must be freed by
55  * calling \c{free()}. This is the case even for zero-length strings.
56  *
57  * The \a next pointer, if not null, will be updated to point to the next item
58  * after this string. If \a value points to the last item, then \a next will be
59  * invalid.
60  *
61  * This function may not run in constant time (it will run in O(n) time on the
62  * number of chunks). It requires constant memory (O(1)) in addition to the
63  * malloc'ed block.
64  *
65  * \note This function does not perform UTF-8 validation on the incoming text
66  * string.
67  *
68  * \sa cbor_value_get_text_string_chunk(), cbor_value_copy_text_string(),

102  size_t *buflen, CborValue *next) {
103  CborError err;
104  cbor_assert(buffer);
105  cbor_assert(buflen);
106  *buflen = SIZE_MAX;

108  if (err)

110 
111  ++*buflen;

113  if (!*buffer) {

117  err = _cbor_value_copy_string(value, *buffer, buflen, next);
