55  for (size_t i = 0; i < pst->num_peers; i++) {
56  peerstore_free_peer(&pst->peers[i]);

89 /*!
90  * @brief used to insert a peer_id public_key pair into our peerstore
91  * @details checks to see if we have a peer before inserting, and all inserts
92  * @details are verified such that we ensure the given public_key matches the peer_id
93  * @details however we should do this after we check if we have the peer to prevent
94  * excessive work
95  * @note if we already have the peer we report true and return early
96  * @todo fix possible race condition with peerstore_have_peer check and the write
97  * lock we claim here
98  * @warning peer_id must be sha256 based
99  * @todo add peer_id and public_key verification
100  * @return Success (not addedd): true
101  * @return Success(added): true
102  * @return Failure: false
103  */
104 bool peerstore_insert_peer(peerstore_t *pst, unsigned char *peer_id,
105  unsigned char *public_key, size_t peer_id_len,
106  size_t public_key_len) {
107 
108  bool ok = false;
109 
110  // perform validation checks to prevent expensive computation if possible
111 
112  /*! @todo this isn't entirely race free, and it is possible for 2 requests to
113  * occur in a row */
114 
115  /*! *@todo enable lru type cache to clean out old entries */
116  if (pst->num_peers == pst->max_peers) {
117  printf("too many peers stored\n");
118  return ok;
119  }
120 
121  /*! @todo dont insert if we already have data for this peer */
122  /*! @todo this is where the race condition lies */
123  if (peerstore_have_peer(pst, peer_id) == true) {
124  return ok;
125  }
126 
127  /*! @note compute the peerID from the public key we are given */
128  /*! @note we then match this with the peerID we are being given */
129  /*! @note this ensures we dont enter any bad data into the peerstore */
130  if (peerstore_validate_peer_id(peer_id, public_key, peer_id_len,
131  public_key_len) == false) {
132  return ok;
133  }
134 
135  pthread_rwlock_wrlock(&pst->mutex);
136 
137  /*! @note this checks to see if we have enough room in the dynamic array */
138  /*! @note if we dont have enough room then we will resize increase space by 2x */
139  /*! @note if that isn't greater than the max peers size. if it is greater */
140  /*! @note then we use the max peers as the limit */
141  /*! @note returns true even if we dint need to resize because we have enough room
142  */
143  bool resized = peerstore_resize_if_needed(pst);
144  if (resized == false) {
145  printf("failed to resize memory slot\n");
146  goto EXIT;
147  }
148 
149  /*! @note right now im not sure if we should store this in the peerstore */
150  /*! @note or use a pointer. i think like this would consume less memory */
151  peer_t pt = {.peer_id_len = peer_id_len, .public_key_len = public_key_len};
152 
153  // allocate memory to hold peer id
154  pt.peer_id = calloc(1, peer_id_len);
155  if (peer_id == NULL) {
156  goto EXIT;
157  }
158 
159  // allocate memory to hold public key
160  pt.public_key = calloc(1, public_key_len);
161  if (pt.public_key == NULL) {
162  free(pt.peer_id);
163  goto EXIT;
164  }
165 
166  // copy peerid and public key data into struct
167  memcpy(pt.peer_id, peer_id, peer_id_len);
168  memcpy(pt.public_key, public_key, public_key_len);
169 
170  // store data in the peerstore and increase count
171  pst->peers[pst->num_peers] = pt;
172  pst->num_peers += 1;
173 
174  ok = true;
175 
176 EXIT:
177 
178  pthread_rwlock_unlock(&pst->mutex);
179 
180  return ok;
181 }

183 /*!
184  * @brief performs a check to see if we need to allocate more memory
185  * @warning caller must take care of locking
186  * @details this is called after every insertion to the peerstore
187  * @details so that the following call will have fresh memory to store data into
188  * @return Success (enough size): true
189  * @return Success (resized ok): true
190  * @return Failure (resized failed): false
191  */
192 bool peerstore_resize_if_needed(peerstore_t *pst) {
193 
194  if (pst->num_peers == pst->peers_size) {
195  /*!
196  * @note set the new peers size to twice the current
197  * @note but if it ends up being more than the max peers we want
198  * @note lets just use the max limit instead
199  */
200  size_t new_peers_size = pst->peers_size * 2;
201 
202  if (new_peers_size > pst->max_peers) {
203  new_peers_size = pst->max_peers;
204  }
205 
206  // realloc and increase available space by 2x (or max peers)
207  pst->peers = realloc(pst->peers, new_peers_size * sizeof(peer_t));
208  if (pst->peers == NULL) {
209  return false;
210  }
211 
212  pst->peers_size = new_peers_size;
213  }
214 
215  return true;
216 }

284 /*!

286  * @details is responsible for allocating memory and initializing the mutex

288  * @details and will double this whenever the max is reached to prevent repeat

292  peerstore_t *pst = calloc(1, sizeof(peerstore_t));

294  return NULL;
295  }
296  pst->peers = calloc(2, sizeof(peer_t));
297  if (pst->peers == NULL) {

299  return NULL;

302  pst->max_peers = max_peers;
