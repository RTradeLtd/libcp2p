36  printf("failed to malloc command_object\n");

52  return pcobj;

70 /*! @brief checks to see if we have a command named according to run and executes it
71  */
72 int execute(command_object *self, char *run) {
73  for (int i = 0; i < self->command_count; i++) {
74  if (strcmp(self->commands[i]->name, run) == 0) {
75  self->commands[i]->callback(self->argc, self->argv);
76  print_colored(COLORS_GREEN, "[info] command executed successfully\n");
77  return 0;
78  }
79  }
80  print_colored(COLORS_RED, "[error] failed to execute command\n");
81  return -1;
82 }
