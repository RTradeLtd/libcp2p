74 /*!
75  * @brief example function used to showcase how you can handle connections
76  * @note in general should accept a conn_handle_data_t type but this is implementation define
77 */
78 void example_task_func_tcp(void *data) {
79  conn_handle_data_t *hdata = (conn_handle_data_t *)data;
80  char buffer[2048];
81  int rc = read(hdata->conn->socket_number, buffer, 2048);
82  switch (rc) {
83  case 0:
84  hdata->srv->thl->log(hdata->srv->thl, 0, "client disconnected", LOG_LEVELS_DEBUG);
85  goto EXIT;
86  case -1:
87  hdata->srv->thl->logf(hdata->srv->thl, 0, LOG_LEVELS_ERROR, "error encountered during read %s", strerror(errno));
88  goto EXIT;
89  default:
90  // connection was successful and we read some data
91  goto EXIT;
92  }
93  send(hdata->conn->socket_number, buffer, (size_t)rc, 0);
94  /*! @todo figure out proper close procedures
95  */
96 EXIT:
97  close(hdata->conn->socket_number);
98  free(hdata->conn);
99  free(hdata);
100 }
