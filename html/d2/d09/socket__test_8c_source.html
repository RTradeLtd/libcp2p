53 void test_new_socket_server(void **state) {
54  ecdsa_private_key_t *server1_pk = new_ecdsa_private_key();
55  ecdsa_private_key_t *server2_pk = new_ecdsa_private_key();
56 
57  int rc = libp2p_crypto_ecdsa_private_key_save(server1_pk, "server1.pem");
58  assert(rc == 0);
59  rc = libp2p_crypto_ecdsa_private_key_save(server2_pk, "server2.pem");
60  assert(rc == 0);
61 
62  // start server 1
63  thread_logger *thl1 = new_thread_logger(true);
64  socket_server_config_t *config1 = new_socket_server_config(1);
65  config1->max_connections = 100;
66  config1->num_threads = 6;
67  config1->recv_timeout_sec = 3;
68  config1->max_peers = 90;
69  config1->fn_tcp = handle_inbound_rpc;
70  config1->private_key_path = "server1.pem";
71 
72  multi_addr_t *tcp_addr1 = multi_address_new_from_string("/ip4/127.0.0.1/tcp/9090");
73  multi_addr_t *endpoint1 = multi_address_new_from_string("/ip4/127.0.0.1/tcp/9090");
74  config1->addrs[0] = tcp_addr1;
75  config1->num_addrs = 1;
76 
77  SOCKET_OPTS opts[2] = {REUSEADDR, NOBLOCK};
78 
79  // .num_threads = 6, .fn_tcp = example_task_func_tcp, .fn_udp = example_task_func_udp };
80  socket_server_t *server1 = new_socket_server(thl1, config1, opts, 2);
81  assert(server1 != NULL);
82  assert(server1->pstore->max_peers == 90);
83  free_socket_server_config(config1);
84 
85  thpool_add_work(server1->thpool, start_socker_server_wrapper, server1);
86 
87  // start server2
88  thread_logger *thl2 = new_thread_logger(true);
89  socket_server_config_t *config2 = new_socket_server_config(1);
90  config2->max_connections = 100;
91  config2->num_threads = 6;
92  config2->recv_timeout_sec = 3;
93  config2->max_peers = 0; // use to trigger default max peers handling
94  config2->fn_tcp = handle_inbound_rpc;
95  config2->private_key_path = "server2.pem";
96 
97  multi_addr_t *tcp_addr2 = multi_address_new_from_string("/ip4/127.0.0.1/tcp/9091");
98  multi_addr_t *endpoint2 = multi_address_new_from_string("/ip4/127.0.0.1/tcp/9091");
99  config2->addrs[0] = tcp_addr2;
100  config2->num_addrs = 1;
101 
102  // .num_threads = 6, .fn_tcp = example_task_func_tcp, .fn_udp = example_task_func_udp };
103  socket_server_t *server2 = new_socket_server(thl2, config2, opts, 2);
104  assert(server2 != NULL);
105 
106  // verify max peers
107  assert(server2->pstore->max_peers == 100);
108 
109  free_socket_server_config(config2);
110 
111  thpool_add_work(server2->thpool, start_socker_server_wrapper, server2);
112 
113  // give time for the last server to start
114  sleep(1);
115 
116  message_hello_t *send_msg_hello = new_server_message_hello_t(server2);
117  assert(send_msg_hello != NULL);
118 
119  message_t *send_msg = message_hello_t_to_message_t(send_msg_hello, true);
120  assert(send_msg != NULL);
121 
122  rc = socket_server_send(server2, endpoint1, send_msg);
123  assert(rc == 0);
124 
125  // give more than enough time for hello protocol exchange to complete
126  sleep(5);
127  // assert peerstore stuff
128  assert(server1->pstore->num_peers == 1);
129  assert(server2->pstore->num_peers == 1);
130 
131  // TODO: assert peerstore contains expected entries
132 
133  // let stuff spin for a bit
134  sleep(5);
135 
136  signal_shutdown();
137 
138  multi_address_free(endpoint1);
139  multi_address_free(endpoint2);
140  free_message_t(send_msg);
141  free_message_hello_t(send_msg_hello);
142  libp2p_crypto_ecdsa_free(server1_pk);
143  libp2p_crypto_ecdsa_free(server2_pk);
144 
145  sleep(1);
146 
147  free_socket_server(server1);
148  free_socket_server(server2);
149 }

152 void test_cbor_message_t_encoding(void **state) {
153  message_t *msg = calloc(1, sizeof(message_t));
154  assert(msg != NULL);
155  msg->data = calloc(1, 6);
156  assert(msg->data != NULL);
157 
158  msg->data[0] = 'h';
159  msg->data[1] = 'e';
160  msg->data[2] = 'l';
161  msg->data[3] = 'l';
162  msg->data[4] = 'o';
163  msg->data[5] = '\0';
164  msg->len = 6;
165  msg->type = MESSAGE_WANT_PEER_ID;
166 
167  cbor_encoded_data_t *cbdata = cbor_encode_message_t(msg);
168  assert(cbdata != NULL);
169  assert(cbdata->len == 10);
170 
171  message_t *ret_msg = cbor_decode_message_t(cbdata);
172  assert(ret_msg != NULL);
173  assert(ret_msg->len == 6);
174  assert(ret_msg->type == MESSAGE_WANT_PEER_ID);
175  assert(
176  strcmp(
177  (char *)ret_msg->data,
178  (char *)msg->data
179  ) == 0
180  );
181  assert(
182  memcmp(
183  ret_msg->data,
184  msg->data,
185  msg->len
186  ) == 0
187  );
188  assert(
189  memcmp(
190  ret_msg->data,
191  msg->data,
192  ret_msg->len
193  ) == 0
194  );
195  free_cbor_encoded_data(cbdata);
196  free_message_t(msg);
197  free_message_t(ret_msg);
198 }
