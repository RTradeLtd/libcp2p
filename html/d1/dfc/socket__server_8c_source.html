45 /*!
46  * @brief used to create a TCP/UDP socket server ready to accept connections
47  * @param thl an instance of a thread_logger
48  * @param config the configuration settings used for the tcp/udp server
49  * @return Success: pointer to a socket_server_t instance
50  * @return Failure: NULL pointer
51  */
52 socket_server_t *new_socket_server(thread_logger *thl,
53  socket_server_config_t config) {
54 
55  SOCKET_OPTS opts[2] = {REUSEADDR, NOBLOCK};
56 
57  addr_info tcp_hints;
58  addr_info udp_hints;
59 
60  int max_socket_num = 0;
61  int rc = 0;
62 
63  fd_set grouped_socket_set;
64  fd_set tcp_socket_set;
65  fd_set udp_socket_set;
66 
67  // initialize the file descriptor groups
68  FD_ZERO(&tcp_socket_set);
69  FD_ZERO(&udp_socket_set);
70  FD_ZERO(&grouped_socket_set);
71 
72  char ip[1024];
73  char cport[7];
74 
75  for (int i = 0; i < config.num_addrs; i++) {
76 
77  // zero ip and cport, overwriting previous data
78  memset(ip, 0, 1024);
79  memset(cport, 0, 7);
80 
81  // get the ip address associated with the multiaddr
82  rc = multi_address_get_ip_address(config.addrs[i], ip);
83  if (rc != 1) {
84  thl->log(thl, 0, "failed to get ip address from multiaddr",
85  LOG_LEVELS_ERROR);
86  goto EXIT;
87  }
88 
89  // get the port for the address
90  int port = multi_address_get_ip_port(config.addrs[i]);
91  if (port == -1) {
92  thl->log(thl, 0, "failed to get ip port from multiaddr",
93  LOG_LEVELS_ERROR);
94  goto EXIT;
95  }
96  // store port number as a char *
97  sprintf(cport, "%i", port);
98 
99  bool is_tcp = false;
100  bool is_udp = false;
101 
102  // determine whether or not we support tcp and udp
103  if (strstr(config.addrs[i]->string, "/tcp/") != NULL) {
104  is_tcp = true;
105  }
106  if (strstr(config.addrs[i]->string, "/udp/") != NULL) {
107  is_udp = true;
108  }
109 
110  // if the multiaddr has neither a tcp or udp protocol, exit
111  if (is_tcp == false && is_udp == false) {
112  thl->log(thl, 0, "invalid multi_address provided", LOG_LEVELS_ERROR);
113  goto EXIT;
114  }
115 
116  // handle a tcp multi_address
117  if (is_tcp) {
118 
119  addr_info *tcp_bind_address = NULL;
120 
121  memset(&tcp_hints, 0, sizeof(tcp_hints));
122  tcp_hints.ai_family = AF_INET;
123  tcp_hints.ai_socktype = SOCK_STREAM;
124  /*! @warning support non wildcard
125  * @todo support non wildcard
126  */
127  tcp_hints.ai_flags = AI_PASSIVE;
128 
129  rc = getaddrinfo(ip, cport, &tcp_hints, &tcp_bind_address);
130  if (rc != 0) {
131  thl->log(thl, 0, "failed to get tcp addr info", LOG_LEVELS_ERROR);
132  goto EXIT;
133  }
134 
135  int tcp_socket_num =
136  get_new_socket(thl, tcp_bind_address, opts, 2, false);
137  if (tcp_socket_num == -1) {
138  thl->log(thl, 0, "failed to get new tcp socket", LOG_LEVELS_ERROR);
139  goto EXIT;
140  }
141 
142  listen(tcp_socket_num, config.max_connections);
143  if (errno != 0) {
144  thl->logf(thl, 0, LOG_LEVELS_ERROR,
145  "failed to start listening on tcp socket with error %s",
146  strerror(errno));
147  goto EXIT;
148  }
149 
150  FD_SET(tcp_socket_num, &tcp_socket_set);
151  FD_SET(tcp_socket_num, &grouped_socket_set);
152 
153  if (tcp_socket_num > max_socket_num) {
154  max_socket_num = tcp_socket_num;
155  }
156 
157  free(tcp_bind_address);
158  }
159 
160  // handle a udp multi_address
161  if (is_udp) {
162 
163  addr_info *udp_bind_address = NULL;
164 
165  memset(&udp_hints, 0, sizeof(udp_hints));
166  udp_hints.ai_family = AF_INET;
167  udp_hints.ai_socktype = SOCK_DGRAM;
168  /*! @warning support non wildcard
169  * @todo support non wildcard
170  */
171  udp_hints.ai_flags = AI_PASSIVE;
172 
173  rc = getaddrinfo(ip, cport, &udp_hints, &udp_bind_address);
174  if (rc != 0) {
175  thl->log(thl, 0, "failed to get udp addr info", LOG_LEVELS_ERROR);
176  goto EXIT;
177  }
178 
179  int udp_socket_num =
180  get_new_socket(thl, udp_bind_address, opts, 2, false);
181  if (udp_socket_num == -1) {
182  thl->log(thl, 0, "failed to get new udp socket", LOG_LEVELS_ERROR);
183  goto EXIT;
184  }
185 
186  FD_SET(udp_socket_num, &udp_socket_set);
187  FD_SET(udp_socket_num, &grouped_socket_set);
188 
189  if (udp_socket_num > max_socket_num) {
190  max_socket_num = udp_socket_num;
191  }
192 
193  free(udp_bind_address);
194  }
195  }
196 
197  socket_server_t *server =
198  calloc(sizeof(socket_server_t), sizeof(socket_server_t));
199  if (server == NULL) {
200  thl->log(thl, 0, "failed to calloc socket server", LOG_LEVELS_ERROR);
201  goto EXIT;
202  }
203 
204  // increase max socket number by 1 for select usage
205  max_socket_num += 1;
206 
207  // setup the server
208  server->thpool = thpool_init(config.num_threads);
209  server->max_socket_num = max_socket_num;
210  server->grouped_socket_set = grouped_socket_set;
211  server->tcp_socket_set = tcp_socket_set;
212  server->udp_socket_set = udp_socket_set;
213  server->task_func_tcp = config.fn_tcp;
214  server->task_func_udp = config.fn_udp;
215  server->thl = thl;
216  server->thl->log(server->thl, 0, "initialized server", LOG_LEVELS_INFO);
217  pthread_mutex_init(&shutdown_mutex, NULL);
218 
219  return server;
220 
221 EXIT:
222 
223  for (int i = 0; i < 65536; i++) {
224  if (FD_ISSET(i, &tcp_socket_set)) {
225  close(i);
226  }
227  if (FD_ISSET(i, &udp_socket_set)) {
228  close(i);
229  }
230  }
231 
232  return NULL;
233 }

235 /*! @brief terminates a server and frees up resources associated with it
236  */
237 void free_socket_server(socket_server_t *srv) {
238  srv->thl->log(srv->thl, 0, "closing sockets", LOG_LEVELS_INFO);
239  for (int i = 0; i < srv->max_socket_num; i++) {
240  if (FD_ISSET(i, &srv->tcp_socket_set)) {
241  close(i);
242  }
243  if (FD_ISSET(i, &srv->udp_socket_set)) {
244  close(i);
245  }
246  }
247  srv->thl->log(srv->thl, 0, "waiting for existing tasks to exit",
248  LOG_LEVELS_INFO);
249  thpool_destroy(srv->thpool);
250  srv->thl->log(srv->thl, 0, "all taskes exited, goodbye", LOG_LEVELS_INFO);
251  pthread_mutex_destroy(&shutdown_mutex);
252  clear_thread_logger(srv->thl);
253  free(srv);
254 }

256 /*!
257  * @brief starts the socket server which processes new connections
258  * @details when a new connection is accepted (tcp) OR we can receive data on a udp
259  * socket, the given handle_conn_func is used to process that client connection
260  * @param srv an instance of a socket_server_t that has been initialized through
261  * new_socket_server
262  */
263 void start_socket_server(socket_server_t *srv) {
264  /*!
265  * @todo enable customizable timeout
266  */
267  timeout tmt;
268  tmt.tv_sec = 3;
269  tmt.tv_usec = 0;
270 
271  for (;;) {
272  if (do_shutdown == true) {
273  srv->thl->log(srv->thl, 0, "shutdown signal received, exiting",
274  LOG_LEVELS_INFO);
275  return;
276  }
277 
278  /*! * @note copy the main socket list containing both tcp and udp sockets
279  */
280  fd_set working_copy = srv->grouped_socket_set;
281 
282  int rc = select(srv->max_socket_num, &working_copy, NULL, NULL, &tmt);
283 
284  switch (rc) {
285  case 0:
286  srv->thl->log(srv->thl, 0,
287  "no sockets are ready for processing, sleeping",
288  LOG_LEVELS_DEBUG);
289  sleep(0.50);
290  continue;
291  case -1:
292  srv->thl->logf(srv->thl, 0, LOG_LEVELS_ERROR,
293  "an error occured while running select: %s",
294  strerror(errno));
295  sleep(0.50);
296  return;
297  }
298 
299  /*!
300  * @brief iterate over all known sockets
301  * @note this will likely search a few extra sockets, but the overhead should
302  * be neglibie
303  */
304  for (int i = 0; i < srv->max_socket_num; i++) {
305 
306  // check to see whether or not we have a socket with this number
307  if (FD_ISSET(i, &working_copy)) {
308 
309  // if it is a tcp socket, handle it with the tcp worker func
310  if (FD_ISSET(i, &srv->tcp_socket_set)) {
311 
312  client_conn_t *conn = accept_client_conn(srv, i);
313  if (conn == NULL) {
314  sleep(0.50);
315  continue;
316  }
317 
318  conn_handle_data_t *chdata = calloc(sizeof(conn_handle_data_t),
319  sizeof(conn_handle_data_t));
320  if (chdata == NULL) {
321  close(conn->socket_number);
322  free(conn->address);
323  free(conn);
324  sleep(0.50);
325  continue;
326  }
327  chdata->srv = srv;
328  chdata->conn = conn;
329 
330  thpool_add_work(srv->thpool, srv->task_func_tcp, chdata);
331  }
332 
333  // if it is a udp socket, handle it with the udp worker func
334  if (FD_ISSET(i, &srv->udp_socket_set)) {
335 
336  client_conn_t *conn =
337  calloc(sizeof(client_conn_t), sizeof(client_conn_t));
338  if (conn == NULL) {
339  srv->thl->log(srv->thl, 0, "failed to calloc client_t",
340  LOG_LEVELS_ERROR);
341  sleep(0.50);
342  continue;
343  }
344  conn->socket_number = i;
345 
346  conn_handle_data_t *chdata = calloc(sizeof(conn_handle_data_t),
347  sizeof(conn_handle_data_t));
348  if (chdata == NULL) {
349  free(conn);
350  srv->thl->log(srv->thl, 0, "failed to calloc client_t",
351  LOG_LEVELS_ERROR);
352  sleep(0.50);
353  continue;
354  }
355 
356  chdata->srv = srv;
357  chdata->conn = conn;
358 
359  thpool_add_work(srv->thpool, srv->task_func_udp, chdata);
360  }
361  }
362  }
363  // sleep bfor 500 miliseconds before looping again
364  sleep(0.50);
365  }
366 }

378 /*! @brief helper function for accepting client connections
379  * times out new attempts if they take 3 seconds or more
380  * @return Failure: NULL client conn failed
381  * @return Success: non-NULL populated client_conn object
382  */
383 client_conn_t *accept_client_conn(socket_server_t *srv, int socket_num) {
384  // temporary variable for storing socket address
385  sock_addr_storage addr_temp;
386  // set client_len
387  // i tried doing `(sock_addr *)&sizeof(addr_temp)
388  // in the `accept` function call but it didnt work
389  socklen_t client_len = sizeof(addr_temp);
390  int client_socket_num = accept(socket_num, (sock_addr *)&addr_temp, &client_len);
391  // socket number less than 0 is an error
392  if (client_socket_num < 0) {
393  return NULL;
394  }
395  client_conn_t *connection = calloc(sizeof(client_conn_t), sizeof(client_conn_t));
396  if (connection == NULL) {
397  return NULL;
398  }
399  connection->address = &addr_temp;
400  connection->socket_number = client_socket_num;
401  char *addr_inf = get_name_info((sock_addr *)connection->address);
402  srv->thl->logf(srv->thl, 0, LOG_LEVELS_INFO, "accepted new connection: %s",
403  addr_inf);
404  free(addr_inf);
405  return connection;
406 }
