45 /*!
46  * @brief used to create a TCP/UDP socket server ready to accept connections
47  * @param thl an instance of a thread_logger
48  * @param config the configuration settings used for the tcp/udp server
49  * @param sock_opts an array of options to configure the sockets we open with
50  * @param num_opts the number of socket options we are using, providing a number that
51  * does not match the actual number of options is undefined behavior
52  * @return Success: pointer to a socket_server_t instance
53  * @return Failure: NULL pointer
54  * @details once you have used the config and created a new server with
55  * @note once you have used the config and created a new server with
56  */
57 socket_server_t *new_socket_server(thread_logger *thl,
58  socket_server_config_t *config,
59  SOCKET_OPTS sock_opts[], int num_opts) {
60 
61  int max_socket_num = 0;
62  fd_set grouped_socket_set;
63  fd_set tcp_socket_set;
64 
65  // initialize the file descriptor groups
66  FD_ZERO(&tcp_socket_set);
67  FD_ZERO(&grouped_socket_set);
68 
69  // if 0 default to 100
70  if (config->max_peers == 0) {
71  config->max_peers = 100;
72  }
73 
74  if (config->private_key_path == NULL) {
75  thl->log(thl, 0, "no private key path in config", LOG_LEVELS_ERROR);
76  return NULL;
77  }
78 
79  for (int i = 0; i < config->num_addrs; i++) {
80 
81  addr_info *bind_address = multi_addr_to_addr_info(config->addrs[i]);
82  if (bind_address == NULL) {
83  thl->log(thl, 0, "failed to get addr info", LOG_LEVELS_ERROR);
84  goto EXIT;
85  }
86 
87  bool is_tcp = false;
88 
89  if (bind_address->ai_socktype == SOCK_STREAM) {
90  is_tcp = true;
91  }
92 
93  // handle a tcp multi_address
94  if (is_tcp) {
95 
96  int tcp_socket_num =
97  get_new_socket(thl, bind_address, sock_opts, num_opts, false, true);
98  if (tcp_socket_num == -1) {
99  freeaddrinfo(bind_address);
100  thl->log(thl, 0, "failed to get new tcp socket", LOG_LEVELS_ERROR);
101  goto EXIT;
102  }
103 
104  listen(tcp_socket_num, config->max_connections);
105  if (errno != 0) {
106  freeaddrinfo(bind_address);
107  thl->logf(thl, 0, LOG_LEVELS_ERROR,
108  "failed to start listening on tcp socket with error %s",
109  strerror(errno));
110  goto EXIT;
111  }
112 
113  FD_SET(tcp_socket_num, &tcp_socket_set);
114  FD_SET(tcp_socket_num, &grouped_socket_set);
115 
116  if (tcp_socket_num > max_socket_num) {
117  max_socket_num = tcp_socket_num;
118  }
119 
120  if (config->recv_timeout_sec > 0) {
121  set_socket_recv_timeout(tcp_socket_num, config->recv_timeout_sec);
122  }
123 
124  freeaddrinfo(bind_address);
125  }
126  }
127 
128  socket_server_t *server = calloc(1, sizeof(socket_server_t));
129  if (server == NULL) {
130  thl->log(thl, 0, "failed to calloc socket server", LOG_LEVELS_ERROR);
131  goto EXIT;
132  }
133 
134  // increase max socket number by 1 for select usage
135  max_socket_num += 1;
136 
137  // initialize our peerstore
138  server->pstore = peerstore_new_peerstore((size_t)config->max_peers);
139  if (server->pstore == NULL) {
140  thl->log(thl, 0, "failed to create peerstore", LOG_LEVELS_ERROR);
141  free(server);
142  goto EXIT;
143  }
144 
145  // load the private key
146  server->private_key =
147  libp2p_crypto_ecdsa_private_key_from_file(config->private_key_path);
148  if (server->private_key == NULL) {
149  thl->log(thl, 0, "failed to load private key", LOG_LEVELS_ERROR);
150  peerstore_free_peerstore(server->pstore);
151  free(server);
152  goto EXIT;
153  }
154 
155  // load our public key
156  server->public_key = libp2p_crypto_ecdsa_keypair_public(server->private_key);
157  if (server->public_key == NULL) {
158  thl->log(thl, 0, "failed to load public key", LOG_LEVELS_ERROR);
159  peerstore_free_peerstore(server->pstore);
160  libp2p_crypto_ecdsa_free(server->private_key);
161  free(server);
162  goto EXIT;
163  }
164 
165  // load our peer identifier
166  server->peer_id = libp2p_crypto_ecdsa_keypair_peerid(server->private_key);
167  if (server->peer_id == NULL) {
168  thl->log(thl, 0, "failed to load public key", LOG_LEVELS_ERROR);
169  peerstore_free_peerstore(server->pstore);
170  libp2p_crypto_ecdsa_free(server->private_key);
171  libp2p_crypto_public_key_free(server->public_key);
172  free(server);
173  goto EXIT;
174  }
175 
176  // setup remaining server components
177  server->thpool = thpool_init(config->num_threads);
178  server->max_socket_num = max_socket_num;
179  server->grouped_socket_set = grouped_socket_set;
180  server->tcp_socket_set = tcp_socket_set;
181  server->task_func_tcp = config->fn_tcp;
182  server->thl = thl;
183  pthread_mutex_init(&shutdown_mutex, NULL);
184  server->thl->log(server->thl, 0, "initialized server", LOG_LEVELS_INFO);
185 
186  return server;
187 
188 EXIT:
189 
190  for (int i = 0; i < max_socket_num; i++) {
191  if (FD_ISSET(i, &tcp_socket_set)) {
192  close(i);
193  }
194  }
195 
196  return NULL;
197 }

199 /*! @brief terminates a server and frees up resources associated with it
200  */
201 void free_socket_server(socket_server_t *srv) {
202  srv->thl->log(srv->thl, 0, "closing sockets", LOG_LEVELS_INFO);
203 
204  for (int i = 0; i < srv->max_socket_num; i++) {
205 
206  if (FD_ISSET(i, &srv->tcp_socket_set)) {
207 
208  srv->thl->logf(srv->thl, 0, LOG_LEVELS_INFO,
209  "closing tcp socket number %i", i);
210 
211  close(i);
212  }
213  }
214 
215  srv->thl->log(srv->thl, 0, "waiting for existing tasks to exit",
216  LOG_LEVELS_INFO);
217 
218  thpool_destroy(srv->thpool);
219 
220  pthread_mutex_destroy(&shutdown_mutex);
221 
222  peerstore_free_peerstore(srv->pstore);
223 
224  libp2p_crypto_ecdsa_free(srv->private_key);
225 
226  libp2p_crypto_public_key_free(srv->public_key);
227 
228  libp2p_peer_id_free(srv->peer_id);
229 
230  srv->thl->log(srv->thl, 0, "all taskes exited, goodbye", LOG_LEVELS_INFO);
231 
232  clear_thread_logger(srv->thl);
233 
234  free(srv);
235 }

237 /*!
238  * @brief starts the socket server which processes new connections
239  * @details when a new connection is accepted (tcp) OR we can receive data on a udp
240  * socket, the given handle_conn_func is used to process that client connection
241  * @param srv an instance of a socket_server_t that has been initialized through
242  * new_socket_server
243  */
244 void start_socket_server(socket_server_t *srv) {
245  /*!
246  * @todo enable customizable timeout
247  */
248  timeout tmt;
249  tmt.tv_sec = 3;
250  tmt.tv_usec = 0;
251 
252  for (;;) {
253  if (do_shutdown == true) {
254  srv->thl->log(srv->thl, 0, "shutdown signal received, exiting",
255  LOG_LEVELS_INFO);
256  return;
257  }
258 
259  /*! * @note copy the main socket list containing both tcp and udp sockets
260  */
261  fd_set working_copy = srv->grouped_socket_set;
262 
263  int rc = select(srv->max_socket_num, &working_copy, NULL, NULL, &tmt);
264 
265  switch (rc) {
266  case 0:
267  sleep(0.50);
268  continue;
269  case -1:
270  srv->thl->logf(srv->thl, 0, LOG_LEVELS_DEBUG,
271  "an error occured while running select: %s",
272  strerror(errno));
273  sleep(0.50);
274  return;
275  }
276 
277  /*!
278  * @brief iterate over all known sockets
279  * @note this will likely search a few extra sockets, but the overhead should
280  * be neglibie
281  */
282  for (int i = 0; i < srv->max_socket_num; i++) {
283 
284  // check to see whether or not we have a socket with this number
285  if (FD_ISSET(i, &working_copy)) {
286 
287  // if it is a tcp socket, handle it with the tcp worker func
288  if (FD_ISSET(i, &srv->tcp_socket_set)) {
289 
290  client_conn_t *conn = accept_client_conn(srv, i);
291  if (conn == NULL) {
292  sleep(0.50);
293  continue;
294  }
295 
296  conn_handle_data_t *chdata =
297  calloc(1, sizeof(conn_handle_data_t));
298  if (chdata == NULL) {
299  close(conn->socket_number);
300  free(conn);
301  sleep(0.50);
302  continue;
303  }
304  chdata->srv = srv;
305  chdata->conn = conn;
306  thpool_add_work(srv->thpool, srv->task_func_tcp, chdata);
307  }
308  }
309  }
310  // sleep bfor 500 miliseconds before looping again
311  sleep(0.50);
312  }
313 }

325 /*! @brief helper function for accepting client connections
326  * times out new attempts if they take 3 seconds or more
327  * @return Failure: NULL client conn failed
328  * @return Success: non-NULL populated client_conn object
329  */
330 client_conn_t *accept_client_conn(socket_server_t *srv, int socket_num) {
331  // temporary variable for storing socket address
332  sock_addr_storage addr_temp;
333  // set client_len
334  // i tried doing `(sock_addr *)&sizeof(addr_temp)
335  // in the `accept` function call but it didnt work
336  socklen_t client_len = sizeof(addr_temp);
337  int client_socket_num = accept(socket_num, (sock_addr *)&addr_temp, &client_len);
338  // socket number less than 0 is an error
339  if (client_socket_num < 0) {
340  return NULL;
341  }
342  client_conn_t *connection = calloc(1, sizeof(client_conn_t));
343  if (connection == NULL) {
344  return NULL;
345  }
346  connection->socket_number = client_socket_num;
347  if (srv->thl->debug == true) {
348  char *addr_inf = get_name_info((sock_addr *)&addr_temp);
349  srv->thl->logf(srv->thl, 0, LOG_LEVELS_DEBUG, "accepted new connection: %s",
350  addr_inf);
351  free(addr_inf);
352  }
353  return connection;
354 }

386 /*!
387  * @brief handles receiving an rpc message from another peer
388  * @note if you send an inbound message of `5hello` you'll invoke a debug handler to
389  * print to stdout
390  * @warning needs to check to see if the data we are getting is for a tcp or udp
391  * connection
392  * @warning if a tcp connection we need to close the socket (as its the socket
393  * connecting to the client)
394  * @warning if a udp connection we dont close the socket and simply free the
395  * resources
396  */
397 void handle_inbound_rpc(void *data) {
398  conn_handle_data_t *hdata = NULL;
399  hdata = (conn_handle_data_t *)data;
400  if (hdata == NULL) {
401  return;
402  }
403 
404  for (;;) {
405  // check to see if we should exit
406  if (do_shutdown == true) {
407  goto RETURN;
408  }
409 
410  message_t *msg = handle_receive(hdata->srv->thl, hdata->conn->socket_number,
411  MAX_RPC_MSG_SIZE_KB);
412 
413  if (msg == NULL) {
414  goto RETURN;
415  }
416 
417  bool success = false;
418  bool ok = false;
419 
420  switch (msg->type) {
421  case MESSAGE_START_ECDH:
422  success = negotiate_secure_connection(hdata);
423  if (success == false) {
424  hdata->srv->thl->log(
425  hdata->srv->thl, 0,
426  "failed to start secure connection negotiation",
427  LOG_LEVELS_DEBUG);
428  free_message_t(msg);
429  goto RETURN;
430  } else {
431  hdata->srv->thl->log(hdata->srv->thl, 0,
432  "started secure connection negotiation",
433  LOG_LEVELS_DEBUG);
434  }
435  break;
436  case MESSAGE_BEGIN_ECDH:
437  break;
438  case MESSAGE_WANT_PEER_ID:
439  break;
440  case MESSAGE_HAVE_PEER_ID:
441  break;
442  case MESSAGE_WANT_PUB_KEY:
443  break;
444  case MESSAGE_HAVE_PUB_KEY:
445  break;
446  case MESSAGE_HELLO_INT:
447  // fallthrough
448  case MESSAGE_HELLO_FIN:
449  ok = handle_hello_protocol(hdata, msg);
450  if (ok == false) {
451  hdata->srv->thl->log(hdata->srv->thl, 0,
452  "failed to conduct hello protocol exchange",
453  LOG_LEVELS_DEBUG);
454  free_message_t(msg);
455  goto RETURN;
456  } else {
457  hdata->srv->thl->log(
458  hdata->srv->thl, 0,
459  "succesfully conducted hello protocol exchange",
460  LOG_LEVELS_DEBUG);
461  }
462  break;
463  case MESSAGE_ARBITRARY:
464  break;
465  default:
466  break;
467  }
468 
469  free_message_t(msg);
470  }
471 RETURN:
472 
473  hdata->srv->thl->log(hdata->srv->thl, 0, "closing connection", LOG_LEVELS_DEBUG);
474 
475  close(hdata->conn->socket_number);
476  free(hdata->conn);
477  free(hdata);
478 }

489 /*!
490  * @brief used to negotiate a secure connection with the current connection
491  */
492 bool negotiate_secure_connection(conn_handle_data_t *data) {
493  message_t *msg = calloc(1, sizeof(message_t));
494  if (msg == NULL) {
495  return false;
496  }
497 
498  msg->data = calloc(1, 2);
499  if (msg->data == NULL) {
500  free(msg);
501  return false;
502  }
503 
504  msg->type = MESSAGE_BEGIN_ECDH;
505  msg->data[0] = 'o';
506  msg->data[1] = 'k';
507  msg->len = 2;
508 
509  int rc = handle_send(data->srv->thl, data->conn->socket_number, msg);
510 
511  free_message_t(msg);
512 
513  if (rc == -1) {
514  return false;
515  }
516 
517  return true;
518 }

520 /*!
521  * @brief handles receiving a hello protocol message from another peer
522  * @details is responsible for exchanging identification information with a peer
523  * @details and updating our peerstore with the appropriate information
524  */
525 bool handle_hello_protocol(conn_handle_data_t *data, message_t *msg) {
526 
527  // temporary struct to hold message data
528  cbor_encoded_data_t cbdata = {.data = msg->data, .len = msg->len};
529 
530  // decode into the hello protocol message
531  message_hello_t *msg_hello = cbor_decode_hello_t(&cbdata);
532  if (msg_hello == NULL) {
533  data->srv->thl->log(data->srv->thl, 0,
534  "failed to cbor decode message_hello_t",
535  LOG_LEVELS_DEBUG);
536  return false;
537  }
538 
539  // insert the peers information into our peerstore
540  bool ok = peerstore_insert_peer(data->srv->pstore, msg_hello->peer_id,
541  msg_hello->public_key, msg_hello->peer_id_len,
542  msg_hello->public_key_len);
543 
544  free_message_hello_t(msg_hello);
545 
546  if (ok == false) {
547  data->srv->thl->log(data->srv->thl, 0,
548  "failed to insert peer into peerstore",
549  LOG_LEVELS_DEBUG);
550  return ok;
551  } else {
552  data->srv->thl->log(data->srv->thl, 0,
553  "successfully inserted peer into peerstore",
554  LOG_LEVELS_DEBUG);
555  }
556 
557  // if this is MESSAGE_HELLO_FIN it means we dont need to continue the exchange
558  if (msg->type == MESSAGE_HELLO_FIN) {
559  return ok;
560  }
561 
562  message_hello_t *send_msg_hello = new_server_message_hello_t(data->srv);
563  if (send_msg_hello == NULL) {
564  data->srv->thl->log(data->srv->thl, 0,
565  "failed to create message_hello_t from server",
566  LOG_LEVELS_DEBUG);
567  return false;
568  }
569 
570  // send the MESSSAGE_HELLO_FIN message in the exchange
571  message_t *send_msg = message_hello_t_to_message_t(send_msg_hello, false);
572 
573  free_message_hello_t(send_msg_hello);
574 
575  if (send_msg == NULL) {
576  data->srv->thl->log(data->srv->thl, 0, "failed to get message_t",
577  LOG_LEVELS_DEBUG);
578  return false;
579  }
580 
581  // send the data to our peer
582  int rc = handle_send(data->srv->thl, data->conn->socket_number, send_msg);
583 
584  free_message_t(send_msg);
585 
586  if (rc == -1) {
587  data->srv->thl->log(data->srv->thl, 0, "failed to send message",
588  LOG_LEVELS_DEBUG);
589  return false;
590  }
591 
592  return true;
593 }

595 /*!
596  * @brief used for a server to send a message to another server
597  * @details this is a sort of "bi-directional RPC method" whereby a server can send a
598  * @details request to another server acting as a client, but enabling either us
599  * @details or the peer to invoke RPC methods. Essentially it is like
600  * handle_inbound_rpc
601  * @details except it is responsible for sending requests to a remote server, and any
602  * responses
603  * @details from the server are ran through handle_inbound_rpc
604  * @return Success: 0
605  * @return Failure: -1
606  */
607 int socket_server_send(socket_server_t *srv, multi_addr_t *to_address,
608  message_t *msg) {
609 
610  socket_client_t *srv_client = new_socket_client(srv->thl, to_address);
611  if (srv_client == NULL) {
612  return -1;
613  }
614 
615  client_conn_t *conn_data = calloc(1, sizeof(client_conn_t));
616  if (conn_data == NULL) {
617  goto EXIT;
618  }
619 
620  conn_data->socket_number = srv_client->socket_number;
621 
622  conn_handle_data_t *chdata = calloc(1, sizeof(conn_handle_data_t));
623  if (chdata == NULL) {
624  free(conn_data);
625  goto EXIT;
626  }
627 
628  chdata->conn = conn_data;
629  chdata->srv = srv;
630 
631  // send the message to the peer
632  handle_send(srv->thl, srv_client->socket_number, msg);
633 
634  // add work to the threadpool to handle responses to the message we just sent
635  thpool_add_work(srv->thpool, srv->task_func_tcp, chdata);
636 
637  freeaddrinfo(srv_client->peer_address);
638  free(srv_client);
639 
640  // return and dont free up resources as the task func handles this
641  return 0;
642 
643 EXIT:
644 
645  close(srv_client->socket_number);
646 
647  freeaddrinfo(srv_client->peer_address);
648 
649  free(srv_client);
650  return -1;
651 }
