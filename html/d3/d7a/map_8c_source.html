54 static int map_resize(map_base_t *m, int nbuckets) {
55  map_node_t *nodes, *node, *next;
56  map_node_t **buckets;
57  int i;
58  /* Chain all nodes together */
59  nodes = NULL;
60  i = m->nbuckets;
61  while (i--) {
62  node = (m->buckets)[i];
63  while (node) {
64  next = node->next;
65  node->next = nodes;
66  nodes = node;
67  node = next;
68  }
69  }
70  /* Reset buckets */
71  buckets = realloc(m->buckets, sizeof(*m->buckets) * nbuckets);
72  if (buckets != NULL) {
73  m->buckets = buckets;
74  m->nbuckets = nbuckets;
75  }
76  if (m->buckets) {
77  memset(m->buckets, 0, sizeof(*m->buckets) * m->nbuckets);
78  /* Re-add nodes to buckets */
79  node = nodes;
80  while (node) {
81  next = node->next;
82  map_addnode(m, node);
83  node = next;
84  }
85  }
86  /* Return error code if realloc() failed */
87  return (buckets == NULL) ? -1 : 0;
88 }

125 int map_set_(map_base_t *m, const char *key, void *value, int vsize) {
126  int n, err;
127  map_node_t **next, *node;
128  /* Find & replace existing node */
129  next = map_getref(m, key);
130  if (next) {
131  memcpy((*next)->value, value, vsize);
132  return 0;
133  }
134  /* Add new node */
135  node = map_newnode(key, value, vsize);
136  if (node == NULL)
137  goto fail;
138  if (m->nnodes >= m->nbuckets) {
139  n = (m->nbuckets > 0) ? (m->nbuckets << 1) : 1;
140  err = map_resize(m, n);
141  if (err)
142  goto fail;
143  }
144  map_addnode(m, node);
145  m->nnodes++;
146  return 0;
147 fail:
148  if (node)
149  free(node);
150  return -1;
