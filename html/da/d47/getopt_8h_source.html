libcp2p
A libp2p-like protocol with first class support for embedded systems. End goal is a protocol that can run on anything from your fridge, embedded systems, and anything else. Higher level languages bindings will be done via FFI. Heavy WIP

53 struct option {
54  /* name of long option */
55  const char *name;
56  /*
57  * one of no_argument, required_argument, and optional_argument:
58  * whether option takes an argument
59  */
60  int has_arg;
61  /* if not NULL, set *flag to val when option found */
62  int *flag;
63  /* if flag not NULL, value to set *flag to; else return value */
64  int val;
65 };
