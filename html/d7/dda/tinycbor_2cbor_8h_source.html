49 #ifndef SIZE_MAX
50 /* Some systems fail to define SIZE_MAX in <stdint.h>, even though C99 requires it...
51  * Conversion from signed to unsigned is defined in 6.3.1.3 (Signed and unsigned
52  * integers) p2, which says: "the value is converted by repeatedly adding or
53  * subtracting one more than the maximum value that can be represented in the new
54  * type until the value is in the range of the new type." So -1 gets converted to
55  * size_t by adding SIZE_MAX + 1, which results in SIZE_MAX.
56  */
57 #define SIZE_MAX ((size_t)-1)
58 #endif
