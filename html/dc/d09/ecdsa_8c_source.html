libcp2p

58 /*! @brief returns the peerid for the corresponding private key
59  * @warning if hash of public key contains a 0, the output will be incorrect
60  * @warning caller must free returned pointer when no longer needed
61  * @details to get the peerid we take a sha256 hash of the public key file in PEM
62  * format
63  * @details we then generate a multihash of that sha256, and base58 encode it
64  * @param pk a loaded ecdsa_private_key_t instance
65  * @return pointer to an unsigned char peerID
66  */
67 unsigned char *libp2p_crypto_ecdsa_keypair_peerid(ecdsa_private_key_t *pk) {
68  unsigned char *public_key = libp2p_crypto_ecdsa_keypair_public(pk);
69  if (public_key == NULL) {
70  return NULL;
71  }
72  unsigned char *public_key_hash = calloc(sizeof(unsigned char), 64);
73  if (public_key_hash == NULL) {
74  return NULL;
75  }
76  int rc = libp2p_crypto_hashing_sha256(public_key, strlen((char *)public_key),
77  public_key_hash);
78  if (rc != 1) {
79  free(public_key);
80  free(public_key_hash);
81  print_mbedtls_error(rc);
82  return NULL;
83  }
84 
85  unsigned char temp_peer_id[1024];
86  size_t len = (size_t)1024;
87  rc = libp2p_new_peer_id(temp_peer_id, &len, public_key_hash, 32);
88  if (rc != 1) {
89  free(public_key);
90  free(public_key_hash);
91  print_mbedtls_error(rc);
92  return NULL;
93  }
94 
95  size_t tpid_size = strlen((char *)temp_peer_id);
96  unsigned char *peer_id = calloc(sizeof(unsigned char), tpid_size + 2);
97  if (peer_id == NULL) {
98  free(public_key);
99  free(public_key_hash);
100  return NULL;
101  }
102  strcpy((char *)peer_id, (char *)temp_peer_id);
103 
104  free(public_key);
105  free(public_key_hash);
106 
107  return peer_id;
108 }

112  * @note returned value has a null terminating byte at the end

116 unsigned char *libp2p_crypto_ecdsa_keypair_public(ecdsa_private_key_t *pk) {
117  // we use `strcpy` and `strlen` as mbedtls_pk_write_pubkey_pem includes null
118  // terminating byte
119  unsigned char output_buf[1024];
120  int rc = mbedtls_pk_write_pubkey_pem(&pk->pk_ctx, output_buf, 1024);
121  if (rc != 0) {
122  print_mbedtls_error(rc);
123  return NULL;
124  }
125  unsigned char *public_key =
126  malloc(sizeof(unsigned char) * strlen((char *)output_buf) + 1);
127  if (public_key == NULL) {
128  return NULL;
129  }
130  strcpy((char *)public_key, (char *)output_buf);

134 /*!
135  * @brief parses a PEM encoded private key and returns a struct for use
136  * @details the returned mbedtls_*_context in the struct are not suitable for
137  * concurrent use, please access through mutex locks
138  * @param pem_input the PEM encoded ECDSA private key
139  * @return an initialize and populated ecdsa_private_key_t
140  */
141 ecdsa_private_key_t *
142 libp2p_crypto_ecdsa_pem_to_private_key(unsigned char *pem_input) {
143  mbedtls_pk_context pk_context;
144  mbedtls_ecdsa_context ecdsa_context;
145 
146  mbedtls_pk_init(&pk_context);
147  mbedtls_ecdsa_init(&ecdsa_context);
148 
149  int rc = mbedtls_pk_parse_key(&pk_context, pem_input,
150  strlen((char *)pem_input) + 1, NULL, 0);
151  if (rc != 0) {
152  print_mbedtls_error(rc);
153  return NULL;
154  }
155 
156  rc = mbedtls_ecdsa_from_keypair(&ecdsa_context, mbedtls_pk_ec(pk_context));
157  if (rc != 0) {
158  print_mbedtls_error(rc);
159  return NULL;
160  }
161  ecdsa_private_key_t *pk = malloc(sizeof(ecdsa_private_key_t) +
162  sizeof(pk_context) + sizeof(ecdsa_context));
163  if (pk == NULL) {
164  mbedtls_pk_free(&pk_context);
165  mbedtls_ecdsa_free(&ecdsa_context);
166  return NULL;
167  }
168  pk->ecdsa_ctx = ecdsa_context;
169  pk->pk_ctx = pk_context;
170  pthread_mutex_init(&pk->mutex, NULL);
171 
172  return pk;
173 }

175 /*! @brief used to generate an ECDSA keypair
176  * @param private_key the place to store the private key
177  * @param curve the ECC curve to use for key generation
178  * @returns Fail: 0
179  * @returns Success: 1
180  */
181 int libp2p_crypto_ecdsa_keypair_generation(unsigned char *output,
182  mbedtls_ecp_group_id curve) {
183  mbedtls_pk_context ecdsa_key_pair;
184  mbedtls_entropy_context entropy_context;
185  mbedtls_ctr_drbg_context ctr_drb_context;
186  /*! * @todo figure out if we need this
187  */
188  const char *pers = "ecdsa";
189 
190  // initialize mbedtls context(s)
191  mbedtls_ctr_drbg_init(&ctr_drb_context);
192  mbedtls_entropy_init(&entropy_context);
193  mbedtls_pk_init(&ecdsa_key_pair);
194  mbedtls_pk_setup(&ecdsa_key_pair, mbedtls_pk_info_from_type(MBEDTLS_PK_ECKEY));
195 
196  // seed entropy generation
197  int rc = mbedtls_ctr_drbg_seed(&ctr_drb_context, mbedtls_entropy_func,
198  &entropy_context, (const unsigned char *)pers,
199  strlen(pers) != 0);
200  if (rc != 0) {
201  print_mbedtls_error(rc);
202  // TODO(bonedaddy): free up memory
203  return 0;
204  }
205 
206  // generate the actual ecdsa keypair
207  rc = mbedtls_ecdsa_genkey(mbedtls_pk_ec(ecdsa_key_pair), curve,
208  mbedtls_ctr_drbg_random, &ctr_drb_context
209 
210  );
211  if (rc != 0) {
212  print_mbedtls_error(rc);
213  return 0;
214  }
215 
216  // write the private key in PEM format to output
217  rc = mbedtls_pk_write_key_pem(&ecdsa_key_pair, output, 1024);
218  if (rc != 0) {
219  print_mbedtls_error(rc);
220  return 0;
221  }
222 
223  // free up allocated resources
224  mbedtls_pk_free(&ecdsa_key_pair);
225  mbedtls_ctr_drbg_free(&ctr_drb_context);
226  mbedtls_entropy_free(&entropy_context);
227  return 1;
228 }

255  rc = 1;
256  } else {
257  rc = 0;
258  }

264 /*!
265  * @brief loads an ECDSA private key from a file containing a PEM key
266  * @param path the path on disk to a PEM file containing an ECDSA private key
267  * @return Success: pointer to an instance of an ecdsa_private_key_t type
268  * @return Failure: NULL pointer
269  */
270 ecdsa_private_key_t *libp2p_crypto_ecdsa_private_key_from_file(char *path) {
271  int fd = open(path, O_RDONLY, 0640);
272  if (fd <= 0) {
273  printf("failed to open: %s\n", strerror(errno));
274  return NULL;
275  }
276  char pem_buffer[1024];
277  memset(pem_buffer, 0, 1024);
278  int rc = read(fd, pem_buffer, 1024);
279  if (rc == -1) {
280  printf("failed to read from file: %s\n", strerror(errno));
281  return NULL;
282  }
283  close(fd);
284 
285  return libp2p_crypto_ecdsa_pem_to_private_key((unsigned char *)pem_buffer);
286 }
