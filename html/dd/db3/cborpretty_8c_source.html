39 /**
40  * \defgroup CborPretty Converting CBOR to text
41  * \brief Group of functions used to convert CBOR to text form.
42  *
43  * This group contains two functions that can be used to convert a \ref
44  * CborValue object to a text representation. This module attempts to follow
45  * the recommendations from RFC 7049 section 6 "Diagnostic Notation", though it
46  * has a few differences. They are noted below.
47  *
48  * TinyCBOR does not provide a way to convert from the text representation back
49  * to encoded form. To produce a text form meant to be parsed, CborToJson is
50  * recommended instead.
51  *
52  * Either of the functions in this section will attempt to convert exactly one
53  * CborValue object to text. Those functions may return any error documented
54  * for the functions for CborParsing. In addition, if the C standard library
55  * stream functions return with error, the text conversion will return with
56  * error CborErrorIO.
57  *
58  * These functions also perform UTF-8 validation in CBOR text strings. If they
59  * encounter a sequence of bytes that is not permitted in UTF-8, they will return
60  * CborErrorInvalidUtf8TextString. That includes encoding of surrogate points
61  * in UTF-8.
62  *
63  * \warning The output type produced by these functions is not guaranteed to
64  * remain stable. A future update of TinyCBOR may produce different output for
65  * the same input and parsers may be unable to handle it.
66  *
67  * \sa CborParsing, CborToJson, cbor_parser_init()
68  */

70 /**
71  * \addtogroup CborPretty
72  * @{
73  * <h2 class="groupheader">Text format</h2>
74  *
75  * As described in RFC 7049 section 6 "Diagnostic Notation", the format is
76  * largely borrowed from JSON, but modified to suit CBOR's different data
77  * types. TinyCBOR makes further modifications to distinguish different, but
78  * similar values.
79  *
80  * CBOR values are currently encoded as follows:
81  * \par Integrals (unsigned and negative)
82  * Base-10 (decimal) text representation of the value
83  * \par Byte strings:
84  * <tt>"h'"</tt> followed by the Base16 (hex) representation of the binary data,
85  * followed by an ending quote (') \par Text strings: C-style escaped string in
86  * quotes, with C11/C++11 escaping of Unicode codepoints above U+007F. \par Tags: Tag
87  * value, with the tagged value in parentheses. No special encoding of the tagged
88  * value is performed. \par Simple types: <tt>"simple(nn)"</tt> where \c nn is the
89  * simple value \par Null: \c null \par Undefined: \c undefined \par Booleans: \c
90  * true or \c false \par Floating point: If NaN or infinite, the actual words \c NaN
91  * or \c infinite. Otherwise, the decimal representation with as many digits as
92  * necessary to ensure no loss of information. By default, float values are suffixed
93  * by "f" and half-float values suffixed by "f16" (doubles have no suffix). If the
94  * CborPrettyNumericEncodingIndicators flag is active, the values instead are encoded
95  * following the Section 6 recommended encoding indicators: float values are suffixed
96  * with "_2" and half-float with "_1". A decimal point is always present. \par
97  * Arrays: Comma-separated list of elements, enclosed in square brackets ("[" and
98  * "]"). \par Maps: Comma-separated list of key-value pairs, with the key and value
99  * separated by a colon (":"), enclosed in curly braces ("{" and "}").
100  *
101  * The CborPrettyFlags enumerator contains flags to control some aspects of the
102  * encoding:
103  * \par String fragmentation
104  * When the CborPrettyShowStringFragments option is active, text and byte
105  * strings that are transmitted in fragments are shown instead inside
106  * parentheses ("(" and ")") with no preceding number and each fragment is
107  * displayed individually. If a tag precedes the string, then the output
108  * will contain a double set of parentheses. If the option is not active,
109  * the fragments are merged together and the display will not show any
110  * difference from a string transmitted with determinate length.
111  * \par Encoding indicators
112  * Numbers and lengths in CBOR can be encoded in multiple representations.
113  * If the CborPrettyIndicateOverlongNumbers option is active, numbers
114  * and lengths that are transmitted in a longer encoding than necessary
115  * will be indicated, by appending an underscore ("_") to either the
116  * number or the opening bracket or brace, followed by a number
117  * indicating the CBOR additional information: 0 for 1 byte, 1 for 2
118  * bytes, 2 for 4 bytes and 3 for 8 bytes.
119  * If the CborPrettyIndicateIndeterminateLength option is active, maps,
120  * arrays and strings encoded with indeterminate length will be marked by
121  * an underscore after the opening bracket or brace or the string (if not
122  * showing fragments), without a number after it.
123  */
124

144 static inline bool convertToUint64(double v, uint64_t *absolute) {
145  double supremum;
146  v = fabs(v);
147 
148  /* C11 standard section 6.3.1.4 "Real floating and integer" says:
149  *
150  * 1 When a finite value of real floating type is converted to an integer
151  * type other than _Bool, the fractional part is discarded (i.e., the
152  * value is truncated toward zero). If the value of the integral part
153  * cannot be represented by the integer type, the behavior is undefined.
154  *
155  * So we must perform a range check that v <= UINT64_MAX, but we can't use
156  * UINT64_MAX + 1.0 because the standard continues:
157  *
158  * 2 When a value of integer type is converted to a real floating type, if
159  * the value being converted can be represented exactly in the new type,
160  * it is unchanged. If the value being converted is in the range of
161  * values that can be represented but cannot be represented exactly, the
162  * result is either the nearest higher or nearest lower representable
163  * value, chosen in an implementation-defined manner.
164  */
165  supremum = -2.0 * INT64_MIN; /* -2 * (- 2^63) == 2^64 */
166  if (v >= supremum)
167  return false;
168 
169  /* Now we can convert, these two conversions cannot be UB */
170  *absolute = v;
171  return *absolute == v;
172 }
173 #endif

180  size_t n) {

189 /* This function decodes buffer as UTF-8 and prints as escaped UTF-16.
190  * On UTF-8 decoding error, it returns CborErrorInvalidUtf8TextString */
191 static CborError utf8EscapedDump(CborStreamFunction stream, void *out,
192  const void *ptr, size_t n) {
193  const uint8_t *buffer = (const uint8_t *)ptr;
194  const uint8_t *const end = buffer + n;
195  CborError err = CborNoError;
196 
197  while (buffer < end && !err) {
198  uint32_t uc = get_utf8(&buffer, end);
199  if (uc == ~0U)
200  return CborErrorInvalidUtf8TextString;
201 
202  if (uc < 0x80) {
203  /* single-byte UTF-8 */
204  unsigned char escaped = (unsigned char)uc;
205  if (uc < 0x7f && uc >= 0x20 && uc != '\\' && uc != '"') {
206  err = stream(out, "%c", (char)uc);
207  continue;
208  }
209 
210  /* print as an escape sequence */
211  switch (uc) {
212  case '"':
213  case '\\':
214  break;
215  case '\b':
216  escaped = 'b';
217  break;
218  case '\f':
219  escaped = 'f';
220  break;
221  case '\n':
222  escaped = 'n';
223  break;
224  case '\r':
225  escaped = 'r';
226  break;
227  case '\t':
228  escaped = 't';
229  break;
230  default:
231  goto print_utf16;
232  }
233  err = stream(out, "\\%c", escaped);
234  continue;
235  }
236 
237  /* now print the sequence */
238  if (uc > 0xffffU) {
239  /* needs surrogate pairs */
240  err = stream(out, "\\u%04" PRIX32 "\\u%04" PRIX32,
241  (uc >> 10) + 0xd7c0, /* high surrogate */
242  (uc % 0x0400) + 0xdc00);
243  } else {
244  print_utf16:
245  /* no surrogate pair needed */
246  err = stream(out, "\\u%04" PRIX32, uc);
247  }
248  }
249  return err;
250 }

252 static const char *resolve_indicator(const uint8_t *ptr, const uint8_t *end,
253  int flags) {
254  static const char indicators[8][3] = {
255  "_0", "_1", "_2", "_3", "", "", "", /* these are not possible */
256  "_"};
257  const char *no_indicator = indicators[5]; /* empty string */
258  uint8_t additional_information;
259  uint8_t expected_information;
260  uint64_t value;
261  CborError err;
262 
263  if (ptr == end)
264  return NULL; /* CborErrorUnexpectedEOF */
265 
266  additional_information = (*ptr & SmallValueMask);
267  if (additional_information < Value8Bit)
268  return no_indicator;
269 
270  /* determine whether to show anything */
271  if ((flags & CborPrettyIndicateIndeterminateLength) &&
272  additional_information == IndefiniteLength)
273  return indicators[IndefiniteLength - Value8Bit];
274  if ((flags & CborPrettyIndicateOverlongNumbers) == 0)
275  return no_indicator;
276 
277  err = _cbor_value_extract_number(&ptr, end, &value);
278  if (err)
279  return NULL; /* CborErrorUnexpectedEOF */
280 
281  expected_information = Value8Bit - 1;
282  if (value >= Value8Bit)
283  ++expected_information;
284  if (value > 0xffU)
285  ++expected_information;
286  if (value > 0xffffU)
287  ++expected_information;
288  if (value > 0xffffffffU)
289  ++expected_information;
290  return expected_information == additional_information
291  ? no_indicator
292  : indicators[additional_information - Value8Bit];
293 }

301 static CborError container_to_pretty(CborStreamFunction stream, void *out,
302  CborValue *it, CborType containerType,
303  int flags, int recursionsLeft) {
304  const char *comma = "";
305  CborError err = CborNoError;
306 
307  if (!recursionsLeft) {
308  printRecursionLimit(stream, out);
309  return err; /* do allow the dumping to continue */
310  }
311 
312  while (!cbor_value_at_end(it) && !err) {
313  err = stream(out, "%s", comma);
314  comma = ", ";
315 
316  if (!err)
317  err = value_to_pretty(stream, out, it, flags, recursionsLeft);
318 
319  if (containerType == CborArrayType)
320  continue;
321 
322  /* map: that was the key, so get the value */
323  if (!err)
324  err = stream(out, ": ");
325  if (!err)
326  err = value_to_pretty(stream, out, it, flags, recursionsLeft);
327  }
328  return err;
329 }

331 static CborError value_to_pretty(CborStreamFunction stream, void *out, CborValue *it,
332  int flags, int recursionsLeft) {
333  CborError err = CborNoError;
334  CborType type = cbor_value_get_type(it);
335  switch (type) {
336  case CborArrayType:
337  case CborMapType: {
338  /* recursive type */
339  CborValue recursed;
340  const char *indicator = get_indicator(it, flags);
341  const char *space = *indicator ? " " : indicator;
342 
343  err = stream(out, "%c%s%s", type == CborArrayType ? '[' : '{', indicator,
344  space);
345  if (err)
346  return err;
347 
348  err = cbor_value_enter_container(it, &recursed);
349  if (err) {
350  it->ptr = recursed.ptr;
351  return err; /* parse error */
352  }
353  err = container_to_pretty(stream, out, &recursed, type, flags,
354  recursionsLeft - 1);
355  if (err) {
356  it->ptr = recursed.ptr;
357  return err; /* parse error */
358  }
359  err = cbor_value_leave_container(it, &recursed);
360  if (err)
361  return err; /* parse error */
362 
363  return stream(out, type == CborArrayType ? "]" : "}");
364  }
365 
366  case CborIntegerType: {
367  uint64_t val;
368  cbor_value_get_raw_integer(it, &val); /* can't fail */
369 
370  if (cbor_value_is_unsigned_integer(it)) {
371  err = stream(out, "%" PRIu64, val);
372  } else {
373  /* CBOR stores the negative number X as -1 - X
374  * (that is, -1 is stored as 0, -2 as 1 and so forth) */
375  if (++val) { /* unsigned overflow may happen */
376  err = stream(out, "-%" PRIu64, val);
377  } else {
378  /* overflown
379  * 0xffff`ffff`ffff`ffff + 1 =
380  * 0x1`0000`0000`0000`0000 = 18446744073709551616 (2^64) */
381  err = stream(out, "-18446744073709551616");
382  }
383  }
384  if (!err)
385  err = stream(out, "%s", get_indicator(it, flags));
386  break;
387  }
388 
389  case CborByteStringType:
390  case CborTextStringType: {
391  size_t n = 0;
392  const void *ptr;
393  bool showingFragments = (flags & CborPrettyShowStringFragments) &&
394  !cbor_value_is_length_known(it);
395  const char *separator = "";
396  char close = '\'';
397  char open[3] = "h'";
398  const char *indicator = NULL;
399 
400  if (type == CborTextStringType) {
401  close = open[0] = '"';
402  open[1] = '\0';
403  }
404 
405  if (showingFragments) {
406  err = stream(out, "(_ ");
407  if (!err)
408  err = _cbor_value_prepare_string_iteration(it);
409  } else {
410  err = stream(out, "%s", open);
411  }
412 
413  while (!err) {
414  if (showingFragments || indicator == NULL) {
415  /* any iteration, except the second for a non-chunked string */
416  indicator = resolve_indicator(it->ptr, it->parser->end, flags);
417  }
418 
419  err = _cbor_value_get_string_chunk(it, &ptr, &n, it);
420  if (err)
421  return err;
422  if (!ptr)
423  break;
424 
425  if (!err && showingFragments)
426  err = stream(out, "%s%s", separator, open);
427  if (!err)
428  err = (type == CborByteStringType
429  ? hexDump(stream, out, ptr, n)
430  : utf8EscapedDump(stream, out, ptr, n));
431  if (!err && showingFragments) {
432  err = stream(out, "%c%s", close, indicator);
433  separator = ", ";
434  }
435  }
436 
437  if (!err) {
438  if (showingFragments)
439  err = stream(out, ")");
440  else
441  err = stream(out, "%c%s", close, indicator);
442  }
443  return err;
444  }
445 
446  case CborTagType: {
447  CborTag tag;
448  cbor_value_get_tag(it, &tag); /* can't fail */
449  err = stream(out, "%" PRIu64 "%s(", tag, get_indicator(it, flags));
450  if (!err)
451  err = cbor_value_advance_fixed(it);
452  if (!err && recursionsLeft)
453  err = value_to_pretty(stream, out, it, flags, recursionsLeft - 1);
454  else if (!err)
455  printRecursionLimit(stream, out);
456  if (!err)
457  err = stream(out, ")");
458  return err;
459  }
460 
461  case CborSimpleType: {
462  /* simple types can't fail and can't have overlong encoding */
463  uint8_t simple_type;
464  cbor_value_get_simple_type(it, &simple_type);
465  err = stream(out, "simple(%" PRIu8 ")", simple_type);
466  break;
467  }
468 
469  case CborNullType:
470  err = stream(out, "null");
471  break;
472 
473  case CborUndefinedType:
474  err = stream(out, "undefined");
475  break;
476 
477  case CborBooleanType: {
478  bool val;
479  cbor_value_get_boolean(it, &val); /* can't fail */
480  err = stream(out, val ? "true" : "false");
481  break;
482  }
483 
484 #ifndef CBOR_NO_FLOATING_POINT
485  case CborDoubleType: {
486  const char *suffix;
487  double val;
488  int r;
489  uint64_t ival;
490 
491  if (false) {
492  float f;
493  case CborFloatType:
494  cbor_value_get_float(it, &f);
495  val = f;
496  suffix =
497  flags & CborPrettyNumericEncodingIndicators ? "_2" : "f";
498  } else if (false) {
499  uint16_t f16;
500  case CborHalfFloatType:
501 #ifndef CBOR_NO_HALF_FLOAT_TYPE
502  cbor_value_get_half_float(it, &f16);
503  val = decode_half(f16);
504  suffix =
505  flags & CborPrettyNumericEncodingIndicators ? "_1" : "f16";
506 #else
507  (void)f16;
508  err = CborErrorUnsupportedType;
509  break;
510 #endif
511  } else {
512  cbor_value_get_double(it, &val);
513  suffix = "";
514  }
515 
516  if ((flags & CborPrettyNumericEncodingIndicators) == 0) {
517  r = fpclassify(val);
518  if (r == FP_NAN || r == FP_INFINITE)
519  suffix = "";
520  }
521 
522  if (convertToUint64(val, &ival)) {
523  /* this double value fits in a 64-bit integer, so show it as such
524  * (followed by a floating point suffix, to disambiguate) */
525  err = stream(out, "%s%" PRIu64 ".%s", val < 0 ? "-" : "", ival,
526  suffix);
527  } else {
528  /* this number is definitely not a 64-bit integer */
529  err = stream(out, "%." DBL_DECIMAL_DIG_STR "g%s", val, suffix);
530  }
531  break;
532  }
533 #else
534  case CborDoubleType:
535  case CborFloatType:
536  case CborHalfFloatType:
537  err = CborErrorUnsupportedType;
538  break;
539 #endif /* !CBOR_NO_FLOATING_POINT */
540 
541  case CborInvalidType:
542  err = stream(out, "invalid");
543  if (err)
544  return err;
545  return CborErrorUnknownType;
546  }
547 
548  if (!err)
549  err = cbor_value_advance_fixed(it);
550  return err;
551 }
